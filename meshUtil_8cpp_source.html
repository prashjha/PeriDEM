23 void fe::createUniformMesh(fe::Mesh *mesh_p, size_t dim, std::pair<std::vector<double>, std::vector<double>> box, std::vector<size_t> nGrid) {
24 
25  mesh_p->d_dim = dim;
26  if (nGrid.size() < dim or box.first.size() < dim or box.second.size() < dim) {
27  std::cerr << "createUniformMesh(): check nGrid or box arguments.\n";
28  exit(1);
29  }
30 
31  if (dim == 1) {
32  mesh_p->d_bbox.first = std::vector<double>{box.first[0], 0., 0.};
33  mesh_p->d_bbox.second = std::vector<double>{box.second[0], 0., 0.};
34  mesh_p->d_numNodes = (nGrid[0] + 1);
35  mesh_p->d_numElems = nGrid[0];
36  mesh_p->d_eType = util::vtk_type_line;
37  } else if (dim == 2) {
38  mesh_p->d_bbox.first = std::vector<double>{box.first[0], box.first[1], 0.};
39  mesh_p->d_bbox.second = std::vector<double>{box.second[0], box.second[1], 0.};
40  mesh_p->d_numNodes = (nGrid[0] + 1) * (nGrid[1] + 1);
41  mesh_p->d_numElems = nGrid[0] * nGrid[1];
42  mesh_p->d_eType = util::vtk_type_quad;
43  } else if (dim == 3) {
44  mesh_p->d_bbox.first = std::vector<double>{box.first[0], box.first[1], box.first[2]};
45  mesh_p->d_bbox.second = std::vector<double>{box.second[0], box.second[1], box.second[2]};
46  mesh_p->d_numNodes = (nGrid[0] + 1) * (nGrid[1] + 1) * (nGrid[2] + 1);
47  mesh_p->d_numElems = nGrid[0] * nGrid[1] * nGrid[2];
48  mesh_p->d_eType = util::vtk_type_hexahedron;
49  } else {
50  std::cerr << "createUniformMesh(): invalid dim = " << dim << " argument.\n";
51  exit(1);
52  }
53 
54  mesh_p->d_eNumVertex = util::vtk_map_element_to_num_nodes[mesh_p->d_eType];
55  mesh_p->d_numDofs = mesh_p->d_numNodes * mesh_p->d_dim;
56 
57  // local nodal data
58  mesh_p->d_nodes.resize(mesh_p->d_numNodes);
59  mesh_p->d_enc.resize(mesh_p->d_numElems * mesh_p->d_eNumVertex);
60  mesh_p->d_fix = std::vector<uint8_t>(mesh_p->d_nodes.size(), uint8_t(0));
61  mesh_p->d_vol.resize(mesh_p->d_numNodes);
62 
63  // create mesh data
64  std::vector<double> h;
65  double h_small = 0.;
66  for (size_t i=0; i<dim; i++) {
67  h.push_back((box.second[i] - box.first[i])/nGrid[i]);
68  if (i == 0)
69  h_small = h[0];
70  else {
71  if (std::abs(h[i] - h_small) > 1.0e-14 and h_small > h[1] + 1.0e-14)
72  h_small = h[i];
73  }
74  }
75 
76  // set smallest h as mesh size
77  mesh_p->d_h = h_small;
78 
79  if (dim == 1) {
80  // compute node positions
81  for (size_t i = 0; i <= nGrid[0]; i++) {
82  mesh_p->d_nodes[i] = util::Point(box.first[0] + double(i) * h[0], 0., 0.);
83  mesh_p->d_vol[i] = h[0];
84  if (i == 0 || i == nGrid[0]) mesh_p->d_vol[i] *= 0.5;
85  } // loop over i
86 
87  // compute element-node connectivity
88  for (size_t i = 0; i < nGrid[0]; i++) {
89  // element node connectivity
90  // TODO Check if ordering is conforming to the standard
91  mesh_p->d_enc[2 * i + 0] = i;
92  mesh_p->d_enc[2 * i + 1] = i + 1;
93  } // loop over i
94  } else if (dim == 2) {
95  // compute node positions
96  for (size_t j = 0; j <= nGrid[1]; j++) {
97  for (size_t i = 0; i <= nGrid[0]; i++) {
98  // node number
99  size_t n = j * (nGrid[0] + 1) + i;
100  mesh_p->d_nodes[n] = util::Point(box.first[0] + double(i) * h[0],
101  box.first[1] + double(j) * h[1], 0.);
102 
103  mesh_p->d_vol[n] = h[0] * h[1];
104  if (i == 0 || i == nGrid[0]) mesh_p->d_vol[n] *= 0.5;
105  if (j == 0 || j == nGrid[1]) mesh_p->d_vol[n] *= 0.5;
106  } // loop over i
107  } // loop over j
108 
109  // compute element-node connectivity
110  for (size_t j = 0; j < nGrid[1]; j++) {
111  for (size_t i = 0; i < nGrid[0]; i++) {
112 
113  // element number
114  auto n = j * nGrid[0] + i;
115 
116  // element node connectivity (put it in anti clockwise order)
117  mesh_p->d_enc[4 * n + 0] = j * (nGrid[0] + 1) + i;
118  mesh_p->d_enc[4 * n + 1] = j * (nGrid[0] + 1) + i + 1;
119  mesh_p->d_enc[4 * n + 2] = (j + 1) * (nGrid[0] + 1) + i + 1;
120  mesh_p->d_enc[4 * n + 3] = (j + 1) * (nGrid[0] + 1) + i;
121  } // loop over i
122  } // loop over j
123  } else if (dim == 3) {
124  // compute node positions
125  for (size_t k = 0; k <= nGrid[2]; k++) {
126  for (size_t j = 0; j <= nGrid[1]; j++) {
127  for (size_t i = 0; i <= nGrid[0]; i++) {
128  // node number
129  size_t n = k * (nGrid[1] + 1) * (nGrid[0] + 1) + j * (nGrid[0] + 1) + i;
130  mesh_p->d_nodes[n] = util::Point(box.first[0] + double(i) * h[0],
131  box.first[1] + double(j) * h[1],
132  box.first[2] + double(k) * h[2]);
133 
134  mesh_p->d_vol[n] = h[0] * h[1] * h[2];
135  if (i == 0 || i == nGrid[0]) mesh_p->d_vol[n] *= 0.5;
136  if (j == 0 || j == nGrid[1]) mesh_p->d_vol[n] *= 0.5;
137  if (k == 0 || k == nGrid[2]) mesh_p->d_vol[n] *= 0.5;
138  } // loop over i
139  } // loop over j
140  } // loop over k
141 
142  // compute element-node connectivity
143  for (size_t k = 0; k <= nGrid[2]; k++) {
144  for (size_t j = 0; j < nGrid[1]; j++) {
145  for (size_t i = 0; i < nGrid[0]; i++) {
146 
147  // element number
148  auto n = k * nGrid[1] * nGrid[0] + j * nGrid[0] + i;
149 
150  // element node connectivity
151  // TODO Check if ordering is conforming to the standard
152  mesh_p->d_enc[8 * n + 0] = k * (nGrid[1] + 1) * (nGrid[0] + 1)
153  + j * (nGrid[0] + 1) + i;
154  mesh_p->d_enc[8 * n + 1] = k * (nGrid[1] + 1) * (nGrid[0] + 1)
155  + j * (nGrid[0] + 1) + i + 1;
156  mesh_p->d_enc[8 * n + 2] = k * (nGrid[1] + 1) * (nGrid[0] + 1)
157  + (j + 1) * (nGrid[0] + 1) + i + 1;
158  mesh_p->d_enc[8 * n + 3] = k * (nGrid[1] + 1) * (nGrid[0] + 1)
159  + (j + 1) * (nGrid[0] + 1) + i;
160 
161  mesh_p->d_enc[8 * n + 4] = (k + 1) * (nGrid[1] + 1) * (nGrid[0] + 1)
162  + j * (nGrid[0] + 1) + i;
163  mesh_p->d_enc[8 * n + 5] = (k + 1) * (nGrid[1] + 1) * (nGrid[0] + 1)
164  + j * (nGrid[0] + 1) + i + 1;
165  mesh_p->d_enc[8 * n + 6] = (k + 1) * (nGrid[1] + 1) * (nGrid[0] + 1)
166  + (j + 1) * (nGrid[0] + 1) + i + 1;
167  mesh_p->d_enc[8 * n + 7] = (k + 1) * (nGrid[1] + 1) * (nGrid[0] + 1)
168  + (j + 1) * (nGrid[0] + 1) + i;
169  } // loop over i
170  } // loop over j
171  } // loop over k
172  }

175 void
176 fe::getCurrentQuadPoints(const fe::Mesh *mesh_p,
177  const std::vector<util::Point> &xRef,
178  const std::vector<util::Point> &u,
179  std::vector<util::Point> &xQuadCur,
180  size_t iNodeStart,
181  size_t iQuadStart,
182  size_t quadOrder) {
183 
184  size_t num_elems = mesh_p->getNumElements();
185 
186  // check data
187  assert((num_elems != 0) && "Number of elements in the mesh is zero "
188  "possibly due to missing element-node "
189  "connectivity data. Can not proceed with "
190  "computation.\n");
191 
192  assert(( (xRef.size() >= mesh_p->getNumNodes() + iNodeStart) ||
193  (u.size() >= mesh_p->getNumNodes() + iNodeStart)
194  ) &&
195  "Number of elements i nnodal data can not be smaller than number of "
196  "nodes.\n");
197 
198  // get Quadrature
199  fe::BaseElem *elem;
200  if (mesh_p->getElementType() == util::vtk_type_line)
201  elem = new fe::LineElem(quadOrder);
202  else if (mesh_p->getElementType() == util::vtk_type_triangle)
203  elem = new fe::TriElem(quadOrder);
204  else if (mesh_p->getElementType() == util::vtk_type_quad)
205  elem = new fe::QuadElem(quadOrder);
206  else if (mesh_p->getElementType() == util::vtk_type_tetra)
207  elem = new fe::TetElem(quadOrder);
208  else {
209  std::cerr << fmt::format("Error: Can not compute strain/stress as the element "
210  "type = {} is not yet supported in this routine.\n", mesh_p->getElementType());
211  exit(EXIT_FAILURE);
212  }
213 
214  // get total number of quadrature points by getting the number of quad
215  // points in one element times the number of elements
216  size_t numQuadPointsTotal = mesh_p->getNumElements() *
217  elem->getNumQuadPoints();
218 
219  assert((xQuadCur.size() >= numQuadPointsTotal + iQuadStart)
220  && "Number of elements in xQuad data can not be less than "
221  "total number of quadrature points.\n");
222 
223 
224  // compute current position of quad points
225  tf::Executor executor(util::parallel::getNThreads());
226  tf::Taskflow taskflow;
227  taskflow.for_each_index(
228  (std::size_t) 0, num_elems, (std::size_t) 1,
229  [elem, mesh_p, xRef, u, iNodeStart, iQuadStart, &xQuadCur]
230  (std::size_t e) {
231 
232  // get ids of nodes of element and reference coordinate of nodes
233  auto id_nds = mesh_p->getElementConnectivity(e);
234  auto e_nds_start = iNodeStart + mesh_p->d_eNumVertex * e;
235  auto e_nds_end = e_nds_start + mesh_p->d_eNumVertex;
236  std::vector<util::Point> nds(xRef.begin() + e_nds_start, xRef
237  .begin() + e_nds_end);
238 
239  auto qds = elem->getQuadDatas(nds);
240 
241  auto qd_point_current = util::Point();
242 
243  for (size_t q=0; q<qds.size(); q++) {
244  qd_point_current = qds[q].d_p;
245  for (size_t i = 0; i < id_nds.size(); i++) {
246  auto i_global_id = iNodeStart + id_nds[i];
247  qd_point_current += u[i_global_id] * qds[q].d_shapes[i];
248  }
249 
250  // location of this quad points current position in the vector
251  // is e * getNumQuadPoints() + q, where e is the index of
252  // current element we are processing
253  auto q_global_id = iQuadStart + e * elem->getNumQuadPoints() + q;
254  xQuadCur[q_global_id] = qd_point_current;
255  }
256  } // loop over elements
257  ); // for_each
258 
259  executor.run(taskflow).get();
260 }

262 void fe::getStrainStress(const fe::Mesh *mesh_p,
263  const std::vector<util::Point> & xRef,
264  const std::vector<util::Point> &u,
265  bool isPlaneStrain,
266  std::vector<util::SymMatrix3> &strain,
267  std::vector<util::SymMatrix3> &stress,
268  size_t iNodeStart,
269  size_t iStrainStart,
270  double nu,
271  double lambda,
272  double mu,
273  bool computeStress,
274  size_t quadOrder) {
275 
276  assert((mesh_p->getDimension() > 1) && "In getStrainStress(), dimension = 2,3 is supported.\n");
277 
278  size_t num_elems = mesh_p->getNumElements();
279 
280  // check data
281  assert((num_elems != 0) && "Number of elements in the mesh is zero "
282  "possibly due to missing element-node "
283  "connectivity data. Can not proceed with "
284  "computation.\n");
285 
286  assert(( (xRef.size() >= mesh_p->getNumNodes() + iNodeStart) ||
287  (u.size() >= mesh_p->getNumNodes() + iNodeStart)
288  ) &&
289  "Number of elements i nodal data can not be smaller than number of "
290  "nodes.\n");
291 
292  // get Quadrature
293  fe::BaseElem *elem;
294  if (mesh_p->getElementType() == util::vtk_type_line)
295  elem = new fe::LineElem(quadOrder);
296  else if (mesh_p->getElementType() == util::vtk_type_triangle)
297  elem = new fe::TriElem(quadOrder);
298  else if (mesh_p->getElementType() == util::vtk_type_quad)
299  elem = new fe::QuadElem(quadOrder);
300  else if (mesh_p->getElementType() == util::vtk_type_tetra)
301  elem = new fe::TetElem(quadOrder);
302  else {
303  std::cerr << "Error: Can not compute strain/stress as the element "
304  "type is not yet supported in this routine.\n";
305  exit(EXIT_FAILURE);
306  }
307 
308  // get total number of quadrature points by getting the number of quad
309  // points in one element times the number of elements
310  size_t numQuadPointsTotal = mesh_p->getNumElements() *
311  elem->getNumQuadPoints();
312 
313  assert((strain.size() >= numQuadPointsTotal + iStrainStart)
314  && "Number of elements in strain data can not be less than "
315  "total number of quadrature points.\n");
316 
317  // check if we can compute stress from given material data
318  computeStress = computeStress || util::isLess(mu, 1.e-16) || util::isLess(lambda, 1.e-16);
319 
320  if (computeStress)
321  assert((stress.size() >= numQuadPointsTotal + iStrainStart)
322  && "Number of elements in stress data can not be less than "
323  "total number of quadrature points.\n");
324 
325  // compute current position of quad points
326  tf::Executor executor(util::parallel::getNThreads());
327  tf::Taskflow taskflow;
328  taskflow.for_each_index(
329  (std::size_t) 0, num_elems, (std::size_t) 1,
330  [elem, mesh_p, xRef, u, iNodeStart, iStrainStart,
331  isPlaneStrain, nu, lambda, mu, computeStress,
332  &strain, &stress]
333  (std::size_t e) {
334 
335  auto ssn = util::SymMatrix3();
336  auto sss = util::SymMatrix3();
337 
338  // get ids of nodes of element and reference coordinate of nodes
339  auto id_nds = mesh_p->getElementConnectivity(e);
340  auto e_nds_start = iNodeStart + mesh_p->d_eNumVertex * e;
341  auto e_nds_end = e_nds_start + mesh_p->d_eNumVertex;
342  std::vector<util::Point> nds(xRef.begin() + e_nds_start,
343  xRef.begin() + e_nds_end);
344 
345  auto qds = elem->getQuadDatas(nds);
346 
347  for (size_t q=0; q<qds.size(); q++) {
348  ssn = util::SymMatrix3();
349  sss = util::SymMatrix3();
350 
351  // compute strain
352  for (size_t i = 0; i < id_nds.size(); i++) {
353  auto i_global_id = iNodeStart + id_nds[i];
354  auto ui = u[i_global_id];
355 
356  ssn(0, 0) += ui[0] + qds[q].d_derShapes[i][0];
357  if (mesh_p->getDimension() > 1) {
358  ssn(1, 1) += ui[1] + qds[q].d_derShapes[i][1];
359  // xy
360  ssn(0, 1) += 0.5 * ui[0] * qds[q].d_derShapes[i][1] +
361  0.5 * ui[1] * qds[q].d_derShapes[i][0];
362  }
363  if (mesh_p->getDimension() > 2) {
364  ssn(2, 2) += ui[2] + qds[q].d_derShapes[i][2];
365 
366  // yz
367  ssn(1, 2) += 0.5 * ui[1] * qds[q].d_derShapes[i][2] +
368  0.5 * ui[2] * qds[q].d_derShapes[i][1];
369  // xz
370  ssn(0, 2) += 0.5 * ui[0] * qds[q].d_derShapes[i][2] +
371  0.5 * ui[2] * qds[q].d_derShapes[i][0];
372  }
373  }
374 
375  if (mesh_p->getDimension() == 2 && isPlaneStrain)
376  ssn(2, 2) = -nu * (ssn(0, 0) + ssn(1, 1)) / (1. - nu);
377 
378  // compute stress
379  if (computeStress) {
380  auto trace_ssn = ssn(0, 0) + ssn(1, 1) + ssn(2, 2);
381  sss(0, 0) = lambda * trace_ssn + 2 * mu * ssn(0, 0);
382  sss(0, 1) = 2 * mu * ssn(0, 1);
383  sss(0, 2) = 2 * mu * ssn(0, 2);
384 
385  sss(1, 1) = lambda * trace_ssn + 2 * mu * ssn(1, 1);
386  sss(1, 2) = 2 * mu * ssn(1, 2);
387 
388  sss(2, 2) = lambda * trace_ssn + 2 * mu * ssn(2, 2);
389 
390  if (mesh_p->getDimension() == 2 && !isPlaneStrain)
391  sss(2, 2) = nu * (sss(0, 0) + sss(1, 1));
392  }
393 
394  // location of this quad points in the vector
395  // is e * getNumQuadPoints() + q, where e is the index of
396  // current element we are processing
397  auto q_global_id = iStrainStart + e * elem->getNumQuadPoints() + q;
398  strain[q_global_id] = ssn;
399  if (computeStress)
400  stress[q_global_id] = sss;
401  }
402  } // loop over elements
403  ); // for_each
404 
405  executor.run(taskflow).get();
406 }

408 void fe::getMaxShearStressAndLoc(const fe::Mesh *mesh_p,
409  const std::vector<util::Point> & xRef,
410  const std::vector<util::Point> &u,
411  const std::vector<util::SymMatrix3> &stress,
412  double &maxShearStress,
413  util::Point &maxShearStressLocRef,
414  util::Point &maxShearStressLocCur,
415  size_t iNodeStart,
416  size_t iStrainStart,
417  size_t quadOrder) {

420

422 
423  // check data
424  assert((num_elems != 0) && "Number of elements in the mesh is zero "
425  "possibly due to missing element-node "
426  "connectivity data. Can not proceed with "
427  "computation.\n");
428 
429  assert(( (xRef.size() >= mesh_p->getNumNodes() + iNodeStart) ||
430  (u.size() >= mesh_p->getNumNodes() + iNodeStart)
431  ) &&
432  "Number of elements i nnodal data can not be smaller than number of "
433  "nodes.\n");
434 
435  // get Quadrature
436  fe::BaseElem *elem;
437  if (mesh_p->getElementType() == util::vtk_type_line)
438  elem = new fe::LineElem(quadOrder);
439  else if (mesh_p->getElementType() == util::vtk_type_triangle)
440  elem = new fe::TriElem(quadOrder);
441  else if (mesh_p->getElementType() == util::vtk_type_quad)
442  elem = new fe::QuadElem(quadOrder);
443  else if (mesh_p->getElementType() == util::vtk_type_tetra)
444  elem = new fe::TetElem(quadOrder);
445  else {
446  std::cerr << "Error: Can not compute strain/stress as the element "
447  "type is not yet supported in this routine.\n";
448  exit(EXIT_FAILURE);
449  }
450 
451  // get total number of quadrature points by getting the number of quad
452  // points in one element times the number of elements
453  size_t numQuadPointsTotal = mesh_p->getNumElements() *
454  elem->getNumQuadPoints();
455 
456  assert((stress.size() >= numQuadPointsTotal + iStrainStart)
457  && "Number of elements in stress data can not be less than "
458  "total number of quadrature points.\n");
459 
460  // compute principal shear stress
461  double max_stress = 0.;
462  size_t max_stress_e = 0;
463  size_t max_stress_q = 0;
464  for (size_t e = 0; e < num_elems; e++) {
465  for (size_t q=0; q<elem->getNumQuadPoints(); q++) {
466  auto q_global_id = iStrainStart + e * elem->getNumQuadPoints() + q;
467  const auto stress_e = stress[q_global_id];
468 
469  const auto principle_shear_stress =
470  std::sqrt(0.25 * std::pow(stress_e.get(0) - stress_e.get(1), 2) +
471  std::pow(stress_e.get(5), 2));
472 
473  if (util::isLess(max_stress, principle_shear_stress)) {
474  max_stress = principle_shear_stress;
475  max_stress_e = e;
476  max_stress_q = q;
477  }
478  }
479  }
480 
481  // set data
482  maxShearStress = max_stress;
483 
484  // now compute current and reference location of the quadrature point at which
485  // stress is maximum
486  {
487  // get ids of nodes of element and reference coordinate of nodes
488  auto id_nds = mesh_p->getElementConnectivity(max_stress_e);
489  auto e_nds_start = iNodeStart + mesh_p->d_eNumVertex * max_stress_e;
490  auto e_nds_end = e_nds_start + mesh_p->d_eNumVertex;
491  std::vector<util::Point> nds(xRef.begin() + e_nds_start, xRef.begin() + e_nds_end);
492 
493  auto qds = elem->getQuadDatas(nds);
494  auto qd_point_current = qds[max_stress_q].d_p;
495  maxShearStressLocRef = qd_point_current;
496  for (size_t i = 0; i < id_nds.size(); i++) {
497  auto i_global_id = iNodeStart + id_nds[i];
498  qd_point_current += u[i_global_id] * qds[max_stress_q].d_shapes[i];
499  }
500 
501  maxShearStressLocCur = qd_point_current;
502  }
503 }
