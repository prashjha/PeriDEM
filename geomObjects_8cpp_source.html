45bool util::geometry::Line::isInside(const util::Point &x) const {
46
47 double l = this->volume();
48
49 auto da = (d_x2 - d_x1) / l;
50 auto db = x - d_x1;
51 double dot = db * da;
52
53 if (util::isLess(dot, 0.) or util::isGreater(dot, l))
54 return false;
55
56
57 auto dx = db - dot * da;
58
59 return util::isLess(dx.length(), 1.0E-10);
60}

190bool util::geometry::Triangle::isInside(const util::Point &x) const {
191
192 if ((x - d_x).length() > d_r)
193 return false;
194
195 if ((x - d_x).length() < this->inscribedRadius())
196 return true;
197
198 double a = this->volume();
199 double a1 =
200 std::abs(util::triangleArea(x, d_vertices[1], d_vertices[2]));
201 double a2 =
202 std::abs(util::triangleArea(d_vertices[0], x, d_vertices[2]));
203 double a3 =
204 std::abs(util::triangleArea(d_vertices[0], d_vertices[1], x));
205

233 double a1 =

242

245 if (a3 < tol*l)
246 return true;
247
248 return false;
249}

1077bool util::geometry::Cylinder::isInside(const util::Point &x) const {
1078
1079 auto dx = x - d_xBegin;

1091 auto dx_project = dx - dx_dot_xa * d_xa;
1092
1093 return !util::isGreater(dx_project.length(), d_r + 1.0E-12);
1094 }
