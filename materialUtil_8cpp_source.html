21 double computeStateMxI(size_t i, const std::vector<util::Point> &nodes,
22  const std::vector<double> &nodal_vol,
23  const std::vector<std::vector<size_t>> &neighbors,
24  const std::vector<std::vector<float>> &neighbors_sq_dist,
25  const double &mesh_size,
26  const material::Material *material) {
27 
28  double horizon = material->getHorizon();
29  const auto &xi = nodes[i];
30  double m = 0.;
31 
32  // upper and lower bound for volume correction
33  auto check_up = horizon + 0.5 * mesh_size;
34  auto check_low = horizon - 0.5 * mesh_size;
35 
36  size_t k = 0;
37  for (size_t j : neighbors[i]) {
38 
39  const auto &xj = nodes[j];
40  double rji = (xj - xi).length();
41  //double rji = std::sqrt(neighbors_sq_dist[i][k]);
42 
43  if (util::isGreater(rji, horizon) or j == i)
44  continue;
45 
46  // get corrected volume of node j
47  auto volj = nodal_vol[j];
48 
49  if (util::isGreater(rji, check_low))
50  volj *= (check_up - rji) / mesh_size;
51 
52  m += std::pow(rji, 2) * material->getInfFn(rji) * volj;
53 
54  k++;
55  }
56 
57  if (util::isLess(m, 1.0E-18)) {
58  std::ostringstream oss;
59  oss << "Error: Weighted nodal volume = " << m
60  << " should not be too close to zero.\n";
61 
62  oss << "Mesh size = " << mesh_size << "\n";
63  oss << "J = " << material->getInfFn(0.5 * horizon) << "\n";
64  oss << material->printStr(0, 0);
65 
66  std::cout << oss.str();
67 
68  exit(1);
69  }
70  return m;
71 }

73 double computeStateThetaxI(size_t i, const std::vector<util::Point> &nodes,
74  const std::vector<util::Point> &nodes_disp,
75  const std::vector<double> &nodal_vol,
76  const std::vector<std::vector<size_t>> &neighbors,
77  const std::vector<std::vector<float>> &neighbors_sq_dist,
78  const double &mesh_size,
79  const material::Material *material,
80  const geometry::Fracture *fracture,
81  const std::vector<double> &mx) {
82 
83  double horizon = material->getHorizon();
84  const auto &xi = nodes[i];
85  const auto &ui = nodes_disp[i];
86  double m = mx[i];
87  double theta = 0.;
88 
89  // upper and lower bound for volume correction
90  auto check_up = horizon + 0.5 * mesh_size;
91  auto check_low = horizon - 0.5 * mesh_size;
92 
93  size_t k = 0;
94  for (size_t j : neighbors[i]) {
95 
96  const auto &xj = nodes[j];
97  const auto &uj = nodes_disp[j];
98  double rji = (xj - xi).length();
99  //double rji = std::sqrt(neighbors_sq_dist[i][k]); // distance in
100  // reference configuration
101 
102  if (util::isGreater(rji, horizon) or j == i)
103  continue;
104 
105  // get corrected volume of node j
106  auto volj = nodal_vol[j];
107 
108  if (util::isGreater(rji, check_low))
109  volj *= (check_up - rji) / mesh_size;
110 
111  // get bond state
112  double bond_state = fracture->getBondState(i, k) ? 0. : 1.;
113 
114  // get change in bond length
115  auto yi = xi + ui;
116  auto yj = xj + uj;
117  double change_length = (yj - yi).length() - rji;
118 
119  theta += bond_state * rji * change_length *
120  material->getInfFn(rji) * volj;
121 
122  k += 1;
123  }
124 
125  return 3. * theta / m;
126 }

128 double computeHydrostaticStrainI(size_t i, const std::vector<util::Point> &nodes,
129  const std::vector<util::Point> &nodes_disp,
130  const std::vector<double> &nodal_vol,
131  const std::vector<std::vector<size_t>> &neighbors,
132  const std::vector<std::vector<float>> &neighbors_sq_dist,
133  const double &mesh_size,
134  const material::Material *material,
135  const geometry::Fracture *fracture,
136  size_t dim) {
137 
138  double horizon = material->getHorizon();
139  const auto &xi = nodes[i];
140  const auto &ui = nodes_disp[i];
141  double theta = 0.;
142 
143  // upper and lower bound for volume correction
144  auto check_up = horizon + 0.5 * mesh_size;
145  auto check_low = horizon - 0.5 * mesh_size;
146 
147  // get volume of ball
148  double vol_ball = std::pow(horizon, 2) * M_PI;
149  if (dim == 3)
150  vol_ball *= horizon * 4. / 3.;
151 
152  size_t k = 0;
153  for (size_t j : neighbors[i]) {
154 
155  const auto &xj = nodes[j];
156  const auto &uj = nodes_disp[j];
157  double rji = (xj - xi).length();
158  //double rji = std::sqrt(neighbors_sq_dist[i][k]);
159 
160  if (util::isGreater(rji, horizon) or j == i)
161  continue;
162 
163  // get corrected volume of node j
164  auto volj = nodal_vol[j];
165 
166  if (util::isGreater(rji, check_low))
167  volj *= (check_up - rji) / mesh_size;
168 
169  // get bond state
170  double bond_state = fracture->getBondState(i, k) ? 0. : 1.;
171 
172  // get bond strain
173  double Sji = material->getS(xj - xi, uj - ui);
174 
175  theta += bond_state * rji * Sji *
176  material->getInfFn(rji) * volj / vol_ball;
177 
178  k += 1;
179  }
180 
181  return theta;
182 }

184 void updateBondFractureDataI(size_t i, const std::vector<util::Point> &nodes,
185  const std::vector<std::vector<size_t>> &neighbors,
186  const std::vector<util::Point> &nodes_disp,
187  const material::Material *material,
188  geometry::Fracture *fracture) {
189 
190  size_t k = 0;
191  for (size_t j : neighbors[i]) {
192 
193  double s = material->getS(nodes[j] - nodes[i], nodes_disp[j] -
194  nodes_disp[i]);
195  double sc = material->getSc((nodes[j] - nodes[i]).length());
196 
197  // get fracture state, modify, and set

199  if (!fs && util::isGreater(std::abs(s), sc + 1.0e-10))
200  fs = true;
201  fracture->setBondState(i, k, fs);
