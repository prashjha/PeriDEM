148struct BoxPartition {
149
150util::Point d_xc; // centroid of box
151std::pair<util::Point, util::Point> d_box; // two corner points
152double d_r; // radius of circle inscribing box
153std::vector<size_t> d_nodes; // id of nodes in this box
154
155BoxPartition() : d_xc(util::Point()),
156 d_box({util::Point(), util::Point()}),
157 d_r(0.) {}
158
159bool isNear(const BoxPartition &box, const double &tol) const {
160 auto dx = d_xc - box.d_xc;
161 return dx.length() < d_r + box.d_r + tol;
162}
163
164bool isNear(const util::Point &x, const double &tol) const {
165
166 if (util::isLess(x.d_x, d_box.first.d_x - tol) or
167 util::isLess(x.d_y, d_box.first.d_y - tol) or
168 util::isLess(x.d_z, d_box.first.d_z - tol) or
169 util::isGreater(x.d_x, d_box.second.d_x + tol) or
170 util::isGreater(x.d_y, d_box.second.d_y + tol) or
171 util::isGreater(x.d_z, d_box.second.d_z + tol))
172 return false;
173
174 return true;
175}
176
177void addNode(const size_t &i) {
178 for (const auto &j: d_nodes)
179 if (j == i)
180 return;
181
182 d_nodes.push_back(i);
183}
