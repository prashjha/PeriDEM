31struct BoxPartition {
32
33 util::Point d_xc; // centroid of box
34 std::pair<util::Point, util::Point> d_box; // two corner points
35 double d_r; // radius of circle inscribing box
36 std::vector<size_t> d_nodes; // id of nodes in this box
37
38 BoxPartition() : d_xc(util::Point()),
39 d_box({util::Point(), util::Point()}),
40 d_r(0.) {}
41
42 bool isNear(const BoxPartition &box, const double &tol) const {
43 auto dx = d_xc - box.d_xc;
44 return dx.length() < d_r + box.d_r + tol;
45 }
46
47 bool isNear(const util::Point &x, const double &tol) const {
48
49 if (util::isLess(x.d_x, d_box.first.d_x - tol) or
50 util::isLess(x.d_y, d_box.first.d_y - tol) or
51 util::isLess(x.d_z, d_box.first.d_z - tol) or
52 util::isGreater(x.d_x, d_box.second.d_x + tol) or
53 util::isGreater(x.d_y, d_box.second.d_y + tol) or
54 util::isGreater(x.d_z, d_box.second.d_z + tol))
55 return false;
56
57 return true;
58 }
59
60 void addNode(const size_t &i) {
61 for (const auto &j : d_nodes)
62 if (j == i)
63 return;
64
65 d_nodes.push_back(i);
66 }
67};
