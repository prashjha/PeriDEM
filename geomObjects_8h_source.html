47  struct BoxPartition {
48 
49  util::Point d_xc; // centroid of box
50  std::pair<util::Point, util::Point> d_box; // two corner points
51  double d_r; // radius of circle inscribing box
52  std::vector<size_t> d_nodes; // id of nodes in this box
53 
54  BoxPartition() : d_xc(util::Point()),
55  d_box({util::Point(), util::Point()}),
56  d_r(0.) {}
57 
58  bool isNear(const BoxPartition &box, const double &tol) const {
59  auto dx = d_xc - box.d_xc;
60  return dx.length() < d_r + box.d_r + tol;
61  }
62 
63  bool isNear(const util::Point &x, const double &tol) const {
64 
65  if (util::isLess(x.d_x, d_box.first.d_x - tol) or
66  util::isLess(x.d_y, d_box.first.d_y - tol) or
67  util::isLess(x.d_z, d_box.first.d_z - tol) or
68  util::isGreater(x.d_x, d_box.second.d_x + tol) or
69  util::isGreater(x.d_y, d_box.second.d_y + tol) or
70  util::isGreater(x.d_z, d_box.second.d_z + tol))
71  return false;
72 
73  return true;
74  }
75 
76  void addNode(const size_t &i) {
77  for (const auto &j: d_nodes)
78  if (j == i)
79  return;
80 
81  d_nodes.push_back(i);
82  }
