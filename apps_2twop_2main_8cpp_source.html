74  log("twop::Model: Running twop app \n");

76  init();
77 
78  // check if init() successfully created quadrature data which we need for postprocessing
79  {
80  size_t totalQuadPoints = 0;
81  for (auto &p: d_particles) {
82  const auto *particle_mesh_p = p->d_rp_p->getMeshP();
83 
84  // get Quadrature
85  fe::BaseElem *elem;
86  if (particle_mesh_p->getElementType() == util::vtk_type_line)
87  elem = new fe::LineElem(d_modelDeck_p->d_quadOrder);
88  else if (particle_mesh_p->getElementType() == util::vtk_type_triangle)
89  elem = new fe::TriElem(d_modelDeck_p->d_quadOrder);
90  else if (particle_mesh_p->getElementType() == util::vtk_type_quad)
91  elem = new fe::QuadElem(d_modelDeck_p->d_quadOrder);
92  else if (particle_mesh_p->getElementType() == util::vtk_type_tetra)
93  elem = new fe::TetElem(d_modelDeck_p->d_quadOrder);
94  else {
95  std::cerr << fmt::format("Error: Can not compute strain/stress as the element "
96  "type = {} is not yet supported in this routine.\n", particle_mesh_p->getElementType());
97  exit(EXIT_FAILURE);
98  }
99 
100  totalQuadPoints += particle_mesh_p->getNumElements() *
101  elem->getNumQuadPoints();
102  }
103 
104  if (d_xQuadCur.size() != totalQuadPoints
105  or d_strain.size() != totalQuadPoints
106  or d_stress.size() != totalQuadPoints) {
107  std::cerr << "Error: DEMModel::init() did not initialize quadrature data.\n";
108  exit(EXIT_FAILURE);
109  }
110  }
111 
112  integrate();

115  void integrate() override {
116  // perform output at the beginning
117  if (d_n == 0 && d_outputDeck_p->d_performOut) output();
118 
119  // apply initial condition
120  if (d_n == 0) applyInitialCondition();
121 
122  // apply loading
123  computeExternalDisplacementBC();
124  computeExternalForces();
125 
126  for (size_t i = d_n; i < d_modelDeck_p->d_Nt; i++) {
127 
128  if (d_n % 100 == 0)
129  log(fmt::format("twop::Model: time step: {} \n", i));
130 
131  // NOTE: If there is need for different time-stepping scheme, one can
132  // define another function similar to these two below
133  clock_begin = steady_clock::now();
134  integrateStep();
135 
136  if (d_n % 100 == 0)
137  log(fmt::format(
138  " Integration time: {} \n",
139  util::methods::timeDiff(clock_begin, steady_clock::now())));
140 
141  if (d_pDeck_p->d_testName == "two_particle") {
142  // NOTE: The purpose of this app 'twop' is to show that if we have
143  // specific post-processing requirement from the outcome of
144  // simulation -- eg in two-particle test, we may be interested in the
145  // maximum of y-coordinate of top particle to measure the damping
146  // effect or the maximum shear stress -- we wrap the base DEMModel
147  // class as we did here and add a specific post-processing function
148  // like 'ppTwoParticleTest()'.
149  twoParticleTest();
150  }
151 
152  // handle general output
153  if ((d_n % d_outputDeck_p->d_dtOut == 0) &&
154  (d_n >= d_outputDeck_p->d_dtOut) && d_outputDeck_p->d_performOut) {
155  clock_begin = steady_clock::now();
156  output();
157  log(fmt::format(
158  " Output time: {} \n",
159  util::methods::timeDiff(clock_begin, steady_clock::now())));
160  }
161 
162  // check for stop (we may want to terminate the simulation early if the
163  // results are garbage, or if some other criteria is met)
164  // NOTE: this too can be specific to particular application
165  checkStop();
166  } // loop over time steps
167  }

191  void twoParticleTestPenetrationDist() {
192 
193  // get alias for particles
194  const auto &p0 = this->d_particles[0];
195  const auto &p1 = this->d_particles[1];
196 
197  // get penetration distance
198  const auto &xc0 = p0->getXCenter();
199  const auto &xc1 = p1->getXCenter();
200  const double &r = p0->d_geom_p->boundingRadius();
201 
202  const auto &contact = d_cDeck_p->getContact(p0->d_zoneId, p1->d_zoneId);
203  double r_e = r + contact.d_contactR;
204 
205  d_penDist = xc1.dist(xc0) - r_e - r;
206  if (util::isLess(d_penDist, 0.))
207  d_contactAreaRadius =
208  std::sqrt(std::pow(r_e, 2.) - std::pow(r_e + d_penDist, 2.));
209  else if (util::isGreater(d_penDist, 0.)) {
210  d_penDist = 0.;
211  d_contactAreaRadius = 0.;
212  }
213 
214  // get max distance of second particle (i.e. the y-coord of center + radius)
215  d_maxDist = xc1.d_y + p1->d_geom_p->boundingRadius();
216 
217  // compute maximum y coordinate of particle 2
218  double max_y_loc = p1->getXLocal(0).d_y;
219  for (size_t i = 0; i < p1->getNumNodes(); i++)
220  if (util::isLess(max_y_loc, p1->getXLocal(i).d_y))
221  max_y_loc = p1->getXLocal(i).d_y;
222 
223  if (util::isLess(d_maxY, max_y_loc))
224  d_maxY = max_y_loc;
225 
226  log(fmt::format("max y: {} \n", d_maxY));
227 
228  // compute ideal values
229  static int contact_pp_ideal = -1;
230  if (contact_pp_ideal == -1) {
231  double mass = p1->getDensity() * M_PI * std::pow(r, 2.);
232 
233  const auto &mat_data =
234  p1->getMaterial()->computeMaterialProperties(d_modelDeck_p->d_dim);
235 
236  d_contactAreaRadiusIdeal = 3. * mass * std::abs(d_pDeck_p->d_gravity[1]) *
237  2. * r * (1. - std::pow(mat_data.d_nu, 2.)) /
238  (4. * mat_data.d_E);
239  d_contactAreaRadiusIdeal = std::pow(d_contactAreaRadiusIdeal, 1. / 3.);
240 
241  d_maxStressLocRefIdeal = r - 0.48 * d_contactAreaRadiusIdeal;
242 
243  d_maxStressIdeal =
244  0.93 * mass * std::abs(d_pDeck_p->d_gravity[1]) /
245  (2. * M_PI * d_contactAreaRadiusIdeal * d_contactAreaRadiusIdeal);
246 
247  contact_pp_ideal = 0;
248  }
249  }

251  void twoParticleTestMaxShearStress() {
252 
253  // compute maximum shear stress and where it occurs
254  double max_stress_t = 0.;
255  auto max_stress_loc_cur_t = util::Point();
256  auto max_stress_loc_ref_t = util::Point();
257 
258  // compute stress and strain
259  for (auto &p: d_particles) {
260  auto p_mat_data = p->getMaterial()->computeMaterialProperties(
261  p->d_rp_p->getMeshP()->getDimension());
262 
263  fe::getCurrentQuadPoints(p->d_rp_p->getMeshP(), d_xRef, d_u, d_xQuadCur,
264  p->d_globStart, p->d_globQuadStart,
265  d_quadOrder);
266  fe::getStrainStress(p->d_rp_p->getMeshP(), d_xRef, d_u,
267  d_input_p->getMaterialDeck()->d_isPlaneStrain,
268  d_strain, d_stress,
269  p->d_globStart, p->d_globQuadStart,
270  p_mat_data.d_nu, p_mat_data.d_lambda,
271  p_mat_data.d_mu,
272  true, d_quadOrder);
273 
274  double p_max_stress = 0.;
275  auto p_max_stress_loc_cur = util::Point();
276  auto p_max_stress_loc_ref = util::Point();
277  fe::getMaxShearStressAndLoc(p->d_rp_p->getMeshP(), d_xRef, d_u, d_stress,
278  p_max_stress,
279  p_max_stress_loc_ref,
280  p_max_stress_loc_cur,
281  p->d_globStart, p->d_globQuadStart, d_quadOrder);
282 
283  if (util::isGreater(p_max_stress, max_stress_t)) {
284  max_stress_t = p_max_stress;
285  auto p_center_node_id = p->d_globStart + p->d_rp_p->getCenterNodeId();
286  max_stress_loc_ref_t = p_max_stress_loc_ref - d_xRef[p_center_node_id];
287  max_stress_loc_cur_t = p_max_stress_loc_cur - d_x[p_center_node_id];
288  }
289  }
290 
291  d_maxStress = max_stress_t;
292  d_maxStressLocRef = max_stress_loc_ref_t.length();
293  d_maxStressLocCur = max_stress_loc_cur_t.length();
294  }
