70  void integrate() override {
71  // perform output at the beginning
72  if (d_n == 0 && d_outputDeck_p->d_performOut) output();
73 
74  // apply initial condition
75  if (d_n == 0) applyInitialCondition();
76 
77  // apply loading
78  computeExternalDisplacementBC();
79  computeExternalForces();
80 
81  for (size_t i = d_n; i < d_modelDeck_p->d_Nt; i++) {
82 
83  if (d_n % 100 == 0)
84  log(fmt::format("time step: {} \n", i));
85 
86  // NOTE: If there is need for different time-stepping scheme, one can
87  // define another function similar to these two below
88  clock_begin = steady_clock::now();
89  integrateStep();
90 
91  if (d_n % 100 == 0)
92  log(fmt::format(
93  " Integration time: {} \n",
94  util::methods::timeDiff(clock_begin, steady_clock::now())));
95 
96  if (d_pDeck_p->d_testName == "two_particle") {
97  // NOTE: The purpose of this app 'twop' is to show that if we have
98  // specific post-processing requirement from the outcome of
99  // simulation -- eg in two-particle test, we may be interested in the
100  // maximum of y-coordinate of top particle to measure the damping
101  // effect or the maximum shear stress -- we wrap the base DEMModel
102  // class as we did here and add a specific post-processing function
103  // like 'ppTwoParticleTest()'.
104  twoParticleTest();
105  }
106 
107  // handle general output
108  if ((d_n % d_outputDeck_p->d_dtOut == 0) &&
109  (d_n >= d_outputDeck_p->d_dtOut) && d_outputDeck_p->d_performOut) {
110  clock_begin = steady_clock::now();
111  output();
112  log(fmt::format(
113  " Output time: {} \n",
114  util::methods::timeDiff(clock_begin, steady_clock::now())));
115  }
116 
117  // check for stop (we may want to terminate the simulation early if the
118  // results are garbage, or if some other criteria is met)
119  // NOTE: this too can be specific to particular application
120  checkStop();
121  } // loop over time steps
122  }

124  // compute location of maximum shear stress and penetration length
125  void twoParticleTest() {
126 
127  bool continue_dt = false;
128  auto check_dt = d_outputDeck_p->d_dtTestOut;
129  if ((d_n % check_dt == 0) && (d_n >= check_dt))
130  continue_dt = true;
131 
132  if (!continue_dt)
133  return;
134 
135  // get alias for particles
136  const auto &p0 = this->d_particles[0];
137  const auto &p1 = this->d_particles[1];
138 
139  // get penetration distance
140  const auto &xc0 = p0->getCurrentCenter();
141  const auto &xc1 = p1->getCurrentCenter();
142  const double &r = p0->getGeomObjectP()->inscribedRadius();
143 
144  const auto &contact = d_cDeck_p->getContact(p0->d_zoneId, p1->d_zoneId);
145  double r_e = r + contact.d_contactR;
146 
147  d_penDist = xc1.dist(xc0) - r_e - r;
148  if (util::definitelyLessThan(d_penDist, 0.))
149  d_contactAreaRadius =
150  std::sqrt(std::pow(r_e, 2.) - std::pow(r_e + d_penDist, 2.));
151  else if (util::definitelyGreaterThan(d_penDist, 0.)) {
152  d_penDist = 0.;
153  d_contactAreaRadius = 0.;
154  }
155 
156  // get max distance of second particle (i.e. the y-coord of center + radius)
157  double max_dist = xc1.d_y + p1->getGeomObject().inscribedRadius();
158 
159  // compute strain
160  double max_stress_t = 0.;
161  auto max_stress_loc_t = util::Point3();
162  p1->getMaxShearStressAndLoc(5, max_stress_t, max_stress_loc_t);
163 
164  d_maxStress = max_stress_t;
165  d_maxStressLoc = max_stress_loc_t.length();
166 
167  // compute ideal values
168  static int contact_pp_ideal = -1;
169  if (contact_pp_ideal == -1) {
170  double mass = p1->getDensity() * M_PI * std::pow(r, 2.);
171 
172  const auto &mat_data =
173  p1->getMaterial()->computeMaterialProperties(d_modelDeck_p->d_dim);
174 
175  d_contactAreaRadiusIdeal = 3. * mass * std::abs(d_pDeck_p->d_gravity[1]) *
176  2. * r * (1. - std::pow(mat_data.d_nu, 2.)) /
177  (4. * mat_data.d_E);
178  d_contactAreaRadiusIdeal = std::pow(d_contactAreaRadiusIdeal, 1. / 3.);
179 
180  d_maxStressLocIdeal = r - 0.48 * d_contactAreaRadiusIdeal;
181 
182  d_maxStressIdeal =
183  0.93 * mass * std::abs(d_pDeck_p->d_gravity[1]) /
184  (2. * M_PI * d_contactAreaRadiusIdeal * d_contactAreaRadiusIdeal);
185 
186  contact_pp_ideal = 0;
187  }
188 
189  d_ppFile << d_time << ", " << -d_penDist << ", " << d_contactAreaRadius
190  << ", " << d_maxStressLoc << ", " << d_maxStress << ", "
191  << max_dist << ", " << d_contactAreaRadiusIdeal << ", "
192  << d_maxStressLocIdeal << ", " << d_maxStressIdeal << std::flush;
193 
194  // compute maximum y coordinate of particle 2
195  double max_y_loc = p1->getCurrentNode(0).d_y;
196  for (size_t i = 0; i < p1->getNumNodes(); i++)
197  if (util::definitelyLessThan(max_y_loc, p1->getCurrentNode(i).d_y))
198  max_y_loc = p1->getCurrentNode(i).d_y;
199 
200  if (util::definitelyLessThan(d_maxY, max_y_loc))
201  d_maxY = max_y_loc;
202 
203  log(fmt::format("max y: {} \n", d_maxY));
204  }
