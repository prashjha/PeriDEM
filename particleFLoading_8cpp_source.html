37void loading::ParticleFLoading::apply(const double &time,
38 particle::BaseParticle *particle) {
39 if (particle->getType() == "particle")
40 this->applyParticle(time, particle);
41 else if (particle->getType() == "wall")
42 this->applyWall(time, particle);

45void loading::ParticleFLoading::applyParticle(const double &time,
46 particle::BaseParticle *particle) {
47
48 for (size_t s = 0; s < d_bcData.size(); s++) {
49
50 // get alias for bc data
51 const auto &bc = d_bcData[s];
52
53 // check if we need to process this particle
54 if ((bc.d_selectionType == "particle" ||
55 bc.d_selectionType == "region_particle") &&
56 !isInList(particle->getTypeId(), bc.d_pList))
57 continue;
58
59 if (bc.d_selectionType == "wall" || bc.d_selectionType == "region_wall")
60 continue;
61
62 // get bounding box
63 auto box = bc.d_region_p->box();
64
65 // for (size_t i = 0; i < particle->getNumNodes(); i++) {
66 tf::Executor executor(util::parallel::getNThreads());
67 tf::Taskflow taskflow;
68
69 taskflow.for_each_index(
70 (std::size_t) 0, particle->getNumNodes(), (std::size_t) 1, [time, &particle, bc, box, this](std::size_t i) {
71 const auto x = particle->getXRefLocal(i);
72 double fmax = 1.0;
73
74 bool compute_force_i = false;
75 if (bc.d_selectionType == "region" && bc.d_region_p->isInside(x))
76 compute_force_i = true;
77 else if (bc.d_selectionType == "region_particle" &&
78 bc.d_region_p->isInside(x) &&
79 isInList(particle->getTypeId(), bc.d_pList))
80 compute_force_i = true;
81 else if (bc.d_selectionType == "particle" &&
82 isInList(particle->getTypeId(), bc.d_pList))
83 compute_force_i = true;
84
85 if (compute_force_i) {
86
87 // apply spatial function
88 if (bc.d_spatialFnType == "hat_x") {
89 fmax = bc.d_spatialFnParams[0] *
90 util::hatFunction(x.d_x, box.first.d_x,
91 box.second.d_x);
92 } else if (bc.d_spatialFnType == "hat_y") {
93 fmax = bc.d_spatialFnParams[0] *
94 util::hatFunction(x.d_y, box.first.d_y,
95 box.second.d_y);
96 } else if (bc.d_spatialFnType == "sin_x") {
97 double a = M_PI * bc.d_spatialFnParams[0];
98 fmax = bc.d_spatialFnParams[0] * std::sin(a * x.d_x);
99 } else if (bc.d_spatialFnType == "sin_y") {
100 double a = M_PI * bc.d_spatialFnParams[0];
101 fmax = bc.d_spatialFnParams[0] * std::sin(a * x.d_y);
102 } else if (bc.d_spatialFnType == "linear_x") {
103 double a = bc.d_spatialFnParams[0];
104 fmax = bc.d_spatialFnParams[0] * a * x.d_x;
105 } else if (bc.d_spatialFnType == "linear_y") {
106 double a = bc.d_spatialFnParams[0];
107 fmax = bc.d_spatialFnParams[0] * a * x.d_y;
108 }
109
110 // apply time function
111 if (bc.d_timeFnType == "linear")
112 fmax *= time;
113 else if (bc.d_timeFnType == "linear_step")
114 fmax *= util::linearStepFunc(time, bc.d_timeFnParams[1],
115 bc.d_timeFnParams[2]);
116 else if (bc.d_timeFnType == "linear_slow_fast") {
117 if (util::isGreater(time, bc.d_timeFnParams[1]))
118 fmax *= bc.d_timeFnParams[3] * time;
119 else
120 fmax *= bc.d_timeFnParams[2] * time;
121 } else if (bc.d_timeFnType == "sin") {
122 double a = M_PI * bc.d_timeFnParams[1];
123 fmax *= std::sin(a * time);
124 }
125
126 // multiply by the slope
127 fmax *= bc.d_timeFnParams[0];
128
129 auto force_i = util::Point();
130 for (auto d : bc.d_direction) {
131 if (d == 1)
132 force_i.d_x += fmax;
133 else if (d == 2)
134 force_i.d_y += fmax;
135 else if (d == 3)
136 force_i.d_z += fmax;
137 }
138
139 // add force
140 particle->addFLocal(i, force_i);
141 } // if compute force
142 }
143 ); // for_each
144
145 executor.run(taskflow).get();
146 } // loop over bc sets
147}
