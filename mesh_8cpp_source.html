65 void fe::Mesh::createData(const std::string &filename, bool ref_config) {
66 
67  util::io::log("Mesh: Reading element data.\n");
68 
69  int file_type = -1;
70  // find the extension of file and call correct reader
71  if (util::io::getExtensionFromFile(filename) == "csv")
72  file_type = 0;
73  else if (util::io::getExtensionFromFile(filename) == "msh")
74  file_type = 1;
75  else if (util::io::getExtensionFromFile(filename) == "vtu")
76  file_type = 2;
77  else {
78  std::cerr << "Error: Currently only '.csv', '.msg', and '.vtu' "
79  "files are supported for reading mesh.\n";
80  exit(EXIT_FAILURE);
81  }
82 
83  if (d_spatialDiscretization != "finite_difference" and file_type == 0) {
84 
85  std::cerr << "Error: For discretization = " << d_spatialDiscretization
86  << " .vtu or .msh mesh file is required.\n";
87  exit(1);
88  }
89 
90  //
91  bool is_fd = false;
92  if (d_spatialDiscretization == "finite_difference")
93  is_fd = true;
94 
95  // read node and elements
96  if (file_type == 0)
97  rw::reader::readCsvFile(filename, d_dim, &d_nodes, &d_vol);
98  else if (file_type == 1) {
99  rw::reader::readMshFile(filename, d_dim, &d_nodes, d_eType, d_numElems,
100  &d_enc, &d_nec, &d_vol, false);
101  d_encDataPopulated = true;
102  }
103  else if (file_type == 2) {
104  //
105  // old reading of mesh
106  //
107  // rw::reader::readVtuFile(filename, d_dim, &d_nodes, d_eType, d_numElems,
108  // &d_enc, &d_nec, &d_vol, false);
109 
110  //
111  // new reading of mesh
112  // We read the data from file one by one depending on what data we need
113  //
114 
115  // read node
116  rw::reader::readVtuFileNodes(filename, d_dim, &d_nodes, ref_config);
117 
118  // read volume if required
119  bool found_volume_data = false;
120  if (is_fd) {
121  found_volume_data =
122  rw::reader::readVtuFilePointData(filename, "Node_Volume", &d_vol);
123 
124  // try another tag for nodal volume
125  if (!found_volume_data)
126  found_volume_data =
127  rw::reader::readVtuFilePointData(filename, "Volume", &d_vol);
128  }
129 
130  // read element data (only if this is fe simulation or if we need
131  // element-node connectivity data for nodal volume calculation)
132  if (!is_fd || !found_volume_data) {
133  rw::reader::readVtuFileCells(filename, d_dim, d_eType, d_numElems, &d_enc,
134  &d_nec);
135  d_encDataPopulated = true;
136  }
137 
138  // check if file has fixity data
139  rw::reader::readVtuFilePointData(filename, "Fixity", &d_fix);
140  }
141 
142  // compute data from mesh data
143  d_numNodes = d_nodes.size();
144  d_eNumVertex = util::vtk_map_element_to_num_nodes[d_eType];
145  d_numDofs = d_numNodes * d_dim;
146 
147  //
148  // assign default values to fixity
149  //
150  if (d_fix.size() != d_numNodes)
151  d_fix = std::vector<uint8_t>(d_nodes.size(), uint8_t(0));
152 
153  //
154  // compute nodal volume if required
155  //
156  bool compute_vol = false;
157  if (is_fd and d_vol.empty()) compute_vol = true;
158 
159  // if this is weak finite element simulation then check from policy if
160  // volume is to be computed
161  if (d_spatialDiscretization == "weak_finite_element")
162  compute_vol = false;
163 
164  if (compute_vol) {
165  util::io::log("Mesh: Computing nodal volume.\n");
166  computeVol();
167  }
168 
169  //
170  // compute bounding box
171  //
172  computeBBox();
173 
174  // check if we need to compute mesh size
175  //if (deck->d_computeMeshSize)
176  computeMeshSize();
177 
178  // check nodal volume
179  size_t counter = 0;
180  for (const auto &v : d_vol) {
181 
182  if (v < 0.01 * std::pow(d_h, d_dim)) {
183 
184  std::cerr << "Error: Check nodal volume " << v
185  << " is less than " << 0.01 * std::pow(d_h, d_dim)
186  << ", Node = " << counter
187  << " at position = " << d_nodes[counter].printStr() << "\n"
188  << "mesh filename = " << filename << "\n"
189  << printStr() << "\n";
190 
191  exit(1);
192  }
193 
194  counter++;
195  }
196 
197  if (d_needEncData and (!d_encDataPopulated or d_enc.empty()))
198  readElementData(d_filename);
199 }

248 void fe::Mesh::computeVol() {
249 
250  // initialize quadrature data
251  fe::BaseElem *quads;
252  if (d_eType == util::vtk_type_triangle)
253  quads = new fe::TriElem(2);
254  else if (d_eType == util::vtk_type_quad)
255  quads = new fe::QuadElem(2);
256  else if (d_eType == util::vtk_type_tetra)
257  quads = new fe::TetElem(2);
258 
259  // check if we have valid element-node connectivity data for nodal volume
260  // calculations
261  if (d_nec.size() != d_numNodes || d_enc.empty()) {
262  std::cerr << "Error: Can not compute nodal volume for given finite "
263  "element mesh as the element-node connectivity data is "
264  "invalid."
265  << std::endl;
266  }
267 
268  if (false) {
269  print(0, 0);
270  std::cout << "\n-------- Node data ----------\n";
271  std::cout << util::io::printStr(d_nodes, 0) << "\n";
272  std::cout << "\n-------- Element data ----------\n";
273  std::cout << util::io::printStr(d_enc, 0) << "\n";
274  }
275 
276  //
277  // compute nodal volume
278  //
279  d_vol.resize(d_numNodes);
280 
281  tf::Executor executor(util::parallel::getNThreads());
282  tf::Taskflow taskflow;
283 
284  taskflow.for_each_index(
285  (std::size_t) 0, this->d_numNodes, (std::size_t) 1, [this, quads](std::size_t i) {
286  double v = 0.0;
287 
288  for (auto e : this->d_nec[i]) {
289 
290  std::vector<size_t> e_ns = this->getElementConnectivity(e);
291 
292  // locate global node i in local list of element el
293  int loc_i = -1;
294  for (size_t l = 0; l < e_ns.size(); l++)
295  if (e_ns[l] == i)
296  loc_i = l;
297 
298  if (loc_i == -1) {
299  std::cerr << "Error: Check node element connectivity.\n";
300  exit(1);
301  }
302 
303  // get quad data
304  std::vector<util::Point> e_nodes;
305  for (auto k : e_ns)
306  e_nodes.emplace_back(this->d_nodes[k]);
307 
308  // get volume of element
309  double vol = quads->elemSize(e_nodes);
310  double factor = 1.;
311  if (vol < 0.)
312  factor = -1.;
313 
314  std::vector<fe::QuadData> qds = quads->getQuadDatas(e_nodes);
315 
316  // compute V_e and add it to volume
317  for (auto qd : qds)
318  v += qd.d_shapes[loc_i] * factor * qd.d_w;
319  } // loop over elements
320 
321  // update
322  this->d_vol[i] = v;
323  }
324  ); // for_each
325 
326  executor.run(taskflow).get();
327 }

329 void fe::Mesh::computeBBox() {
330  std::vector<double> p1(3,0.);
331  std::vector<double> p2(3,0.);
332  for (const auto& x : d_nodes) {
333  if (util::isLess(x.d_x, p1[0]))
334  p1[0] = x.d_x;
335  if (util::isLess(x.d_y, p1[1]))
336  p1[1] = x.d_y;
337  if (util::isLess(x.d_z, p1[2]))
338  p1[2] = x.d_z;
339  if (util::isLess(p2[0], x.d_x))
340  p2[0] = x.d_x;
341  if (util::isLess(p2[1], x.d_y))
342  p2[1] = x.d_y;
343  if (util::isLess(p2[2], x.d_z))
344  p2[2] = x.d_z;
345  }
346 
347  d_bbox = std::make_pair(p1, p2);
348 }

350 void fe::Mesh::computeMeshSize() {
351 
352  double guess = 0.;
353  if (d_nodes.size() < 2) {
354  d_h = 0.;
355  return;
356  }
357 
358  guess = (d_nodes[0] - d_nodes[1]).length();
359  for (size_t i = 0; i < d_nodes.size(); i++)
360  for (size_t j = 0; j < d_nodes.size(); j++)
361  if (i != j) {
362  double val = d_nodes[i].dist(d_nodes[j]);
363 
364  if (util::isLess(val, 1.0E-12)) {
365 
366  std::cout << "Check nodes are too close = "
367  << util::io::printStr<util::Point>({d_nodes[i],
368  d_nodes[j]})
369  << "\n";
370  std::cout << "Distance = " << val << ", guess = " << guess << "\n";
371  }
372  if (util::isLess(val, guess))
373  guess = val;
374  }
375 
376  d_h = guess;
377 }
