64void fe::Mesh::createData(const std::string &filename, bool ref_config) {
65
66 int file_type = -1;
67
68 // find the extension of file and call correct reader
69 if (filename.substr(filename.find_last_of(".") + 1) == "csv")
70 file_type = 0;
71 if (filename.substr(filename.find_last_of(".") + 1) == "msh")
72 file_type = 1;
73 if (filename.substr(filename.find_last_of(".") + 1) == "vtu")
74 file_type = 2;
75
76 if (d_spatialDiscretization != "finite_difference" and file_type == 0) {
77
78 std::cerr << "Error: For discretization = " << d_spatialDiscretization
79 << " .vtu or .msh mesh file is required.\n";
80 exit(1);
81 }
82
83 //
84 bool is_fd = false;
85 if (d_spatialDiscretization == "finite_difference")
86 is_fd = true;
87
88 //
89 // read node and elements
90 //
91 util::io::log("Mesh: Reading mesh.\n");
92
93 if (file_type == 0)
94 rw::reader::readCsvFile(filename, d_dim, &d_nodes, &d_vol);
95 else if (file_type == 1)
96 rw::reader::readMshFile(filename, d_dim, &d_nodes, d_eType, d_numElems,
97 &d_enc, &d_nec, &d_vol, false);
98 else if (file_type == 2) {
99 //
100 // old reading of mesh
101 //
102 // rw::reader::readVtuFile(filename, d_dim, &d_nodes, d_eType, d_numElems,
103 // &d_enc, &d_nec, &d_vol, false);
104
105 //
106 // new reading of mesh
107 // We read the data from file one by one depending on what data we need
108 //
109
110 // read node
111 rw::reader::readVtuFileNodes(filename, d_dim, &d_nodes, ref_config);
112
113 // read volume if required
114 bool found_volume_data = false;
115 if (is_fd) {
116 found_volume_data =
117 rw::reader::readVtuFilePointData(filename, "Node_Volume", &d_vol);
118
119 // try another tag for nodal volume
120 if (!found_volume_data)
121 found_volume_data =
122 rw::reader::readVtuFilePointData(filename, "Volume", &d_vol);
123 }
124
125 // read element data (only if this is fe simulation or if we need
126 // element-node connectivity data for nodal volume calculation)
127 if (!is_fd || !found_volume_data)
128 rw::reader::readVtuFileCells(filename, d_dim, d_eType, d_numElems, &d_enc,
129 &d_nec);
130
131 // check if file has fixity data
132 rw::reader::readVtuFilePointData(filename, "Fixity", &d_fix);
133 }
134
135 // compute data from mesh data
136 d_numNodes = d_nodes.size();
137 d_eNumVertex = util::vtk_map_element_to_num_nodes[d_eType];
138 d_numDofs = d_numNodes * d_dim;
139
140 //
141 // assign default values to fixity
142 //
143 if (d_fix.size() != d_numNodes)
144 d_fix = std::vector<uint8_t>(d_nodes.size(), uint8_t(0));
145
146 //
147 // compute nodal volume if required
148 //
149 bool compute_vol = false;
150 if (is_fd and d_vol.empty()) compute_vol = true;
151
152 // if this is weak finite element simulation then check from policy if
153 // volume is to be computed
154 if (d_spatialDiscretization == "weak_finite_element")
155 compute_vol = false;
156
157 if (compute_vol) {
158 util::io::log("Mesh: Computing nodal volume.\n");
159 computeVol();
160 }
161
162 //
163 // compute bounding box
164 //
165 computeBBox();
166
167 // check if we need to compute mesh size
168 //if (deck->d_computeMeshSize)
169 computeMeshSize();
170
171 // check nodal volume
172 size_t counter = 0;
173 for (const auto &v : d_vol) {
174
175 if (v < 0.01 * std::pow(d_h, d_dim)) {
176
177 std::cerr << "Error: Check nodal volume " << v
178 << " is less than " << 0.01 * std::pow(d_h, d_dim)
179 << ", Node = " << counter
180 << " at position = " << d_nodes[counter].printStr() << "\n";
181
182 exit(1);
183 }
184
185 counter++;
186 }
187}

189void fe::Mesh::computeVol() {
190
191 // initialize quadrature data
192 fe::BaseElem *quads;
193 if (d_eType == util::vtk_type_triangle)
194 quads = new fe::TriElem(2);
195 else if (d_eType == util::vtk_type_quad)
196 quads = new fe::QuadElem(2);
197 else if (d_eType == util::vtk_type_tetra)
198 quads = new fe::TetElem(2);
199
200 // check if we have valid element-node connectivity data for nodal volume
201 // calculations
202 if (d_nec.size() != d_numNodes || d_enc.empty()) {
203 std::cerr << "Error: Can not compute nodal volume for given finite "
204 "element mesh as the element-node connectivity data is "
205 "invalid."
206 << std::endl;
207 }
208
209 if (false) {
210 print(0, 0);
211 std::cout << "\n-------- Node data ----------\n";
212 std::cout << util::io::printStr(d_nodes, 0) << "\n";
213 std::cout << "\n-------- Element data ----------\n";
214 std::cout << util::io::printStr(d_enc, 0) << "\n";
215 }
216
217 //
218 // compute nodal volume
219 //
220 d_vol.resize(d_numNodes);
221
222 tf::Executor executor(util::parallel::getNThreads());
223 tf::Taskflow taskflow;
224
225 taskflow.for_each_index(
226 (std::size_t) 0, this->d_numNodes, (std::size_t) 1, [this, quads](std::size_t i) {
227 double v = 0.0;
228
229 for (auto e : this->d_nec[i]) {
230
231 std::vector<size_t> e_ns = this->getElementConnectivity(e);
232
233 // locate global node i in local list of element el
234 int loc_i = -1;
235 for (size_t l = 0; l < e_ns.size(); l++)
236 if (e_ns[l] == i)
237 loc_i = l;
238
239 if (loc_i == -1) {
240 std::cerr << "Error: Check node element connectivity.\n";
241 exit(1);
242 }
243
244 // get quad data
245 std::vector<util::Point> e_nodes;
246 for (auto k : e_ns)
247 e_nodes.emplace_back(this->d_nodes[k]);
248
249 // get volume of element
250 double vol = quads->elemSize(e_nodes);
251 double factor = 1.;
252 if (vol < 0.)
253 factor = -1.;
254
255 std::vector<fe::QuadData> qds = quads->getQuadDatas(e_nodes);
256
257 // compute V_e and add it to volume
258 for (auto qd : qds)
259 v += qd.d_shapes[loc_i] * factor * qd.d_w;
260 } // loop over elements
261
262 // update
263 this->d_vol[i] = v;
264 }
265 ); // for_each
266
267 executor.run(taskflow).get();
268}

270void fe::Mesh::computeBBox() {
271 std::vector<double> p1(3,0.);
272 std::vector<double> p2(3,0.);
273 for (const auto& x : d_nodes) {
274 if (util::isLess(x.d_x, p1[0]))
275 p1[0] = x.d_x;
276 if (util::isLess(x.d_y, p1[1]))
277 p1[1] = x.d_y;
278 if (util::isLess(x.d_z, p1[2]))
279 p1[2] = x.d_z;
280 if (util::isLess(p2[0], x.d_x))
281 p2[0] = x.d_x;
282 if (util::isLess(p2[1], x.d_y))
283 p2[1] = x.d_y;
284 if (util::isLess(p2[2], x.d_z))
285 p2[2] = x.d_z;
286 }
287
288 d_bbox = std::make_pair(p1, p2);
289}

291void fe::Mesh::computeMeshSize() {
292
293 double guess = 0.;
294 if (d_nodes.size() < 2) {
295 d_h = 0.;
296 return;
297 }
298
299 guess = (d_nodes[0] - d_nodes[1]).length();
300 for (size_t i = 0; i < d_nodes.size(); i++)
301 for (size_t j = 0; j < d_nodes.size(); j++)
302 if (i != j) {
303 double val = d_nodes[i].dist(d_nodes[j]);
304
305 if (util::isLess(val, 1.0E-12)) {
306
307 std::cout << "Check nodes are too close = "
308 << util::io::printStr<util::Point>({d_nodes[i],
309 d_nodes[j]})
310 << "\n";
311 std::cout << "Distance = " << val << ", guess = " << guess << "\n";
312 }
313 if (util::isLess(val, guess))
314 guess = val;
315 }
316
317 d_h = guess;
318}

323void fe::Mesh::setFixity(const size_t &i, const unsigned int &dof,
324 const bool &flag) {
325
326 // to set i^th bit as true of integer a,
327 // a |= 1UL << (i % 8)
328
329 // to set i^th bit as false of integer a,
330 // a &= ~(1UL << (i % 8))
331
332 flag ? (d_fix[i] |= 1UL << dof) : (d_fix[i] &= ~(1UL << dof));
333}
