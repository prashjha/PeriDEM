341  std::pair<double, double> getBondEF(const double &r, const double &s,
342  bool &fs,
343  const bool &break_bonds) const override {
344 
345  if (break_bonds) {
346  // check if fracture state of the bond need to be updated
347  if (d_irrevBondBreak && !fs &&
348  util::isGreater(std::abs(s), d_factorSc * getSc(r)))
349  fs = true;
350 
351  // if bond is not fractured, return energy and force from nonlinear
352  // potential otherwise return energy of fractured bond, and zero force
353  if (!fs)
354  return std::make_pair(
355  getInfFn(r) * d_C *
356  (1. - std::exp(-d_beta * r * s * s)) / d_invFactor,
357  getInfFn(r) * 4. * s * d_C * d_beta *
358  std::exp(-d_beta * r * s * s) / d_invFactor);
359  else
360  return std::make_pair(d_C / d_invFactor, 0.);
361  } else {
362  return std::make_pair(getInfFn(r) * d_C * d_beta *
363  r * s * s / d_invFactor,
364  getInfFn(r) * 4. * s * d_C *
365  d_beta / d_invFactor);
366  }
367  };

540  void computeParameters(inp::MaterialDeck &deck, const size_t &dim) {
541  //
542  // Need following elastic and fracture properties
543  // 1. E or K
544  // 2. Gc or KIc
545  // For bond-based, Poisson's ratio is fixed to 1/4
546  //
547  if (util::isLess(deck.d_matData.d_E, 0.) &&
548  util::isLess(deck.d_matData.d_K, 0.)) {
549  std::cerr << "Error: Require either Young's modulus E or Bulk modulus K"
550  " to compute the RNP bond-based peridynamic parameters.\n";
551  exit(1);
552  }
553  if (util::isGreater(deck.d_matData.d_E, 0.) &&
554  util::isGreater(deck.d_matData.d_K, 0.)) {
555  std::cout << "Warning: Both Young's modulus E and Bulk modulus K are "
556  "provided.\n";
557  std::cout << "Warning: To compute the RNP bond-based peridynamic "
558  "parameters, we only require one of those.\n";
559  std::cout
560  << "Warning: Selecting Young's modulus to compute parameters.\n";
561  }
562 
563  if (util::isLess(deck.d_matData.d_Gc, 0.) &&
564  util::isLess(deck.d_matData.d_KIc, 0.)) {
565  std::cerr << "Error: Require either critical energy release rate Gc or "
566  "critical stress intensity factor KIc to compute the RNP "
567  "bond-based peridynamic parameters.\n";
568  exit(1);
569  } else if (util::isGreater(deck.d_matData.d_Gc, 0.) &&
570  util::isGreater(deck.d_matData.d_KIc, 0.)) {
571  std::cout << "Warning: Both critical energy release rate Gc and critical "
572  "stress intensity factor KIc are provided.\n";
573  std::cout << "Warning: To compute the RNP bond-based peridynamic "
574  "parameters, we only require one of those.\n";
575  std::cout << "Warning: Selecting critical energy release rate Gc to "
576  "compute parameters.\n";
577  }
578 
579  // set Poisson's ratio to 1/4
580  deck.d_matData.d_nu = 0.25;
581 
582  // compute E if not provided or K if not provided
583  if (deck.d_matData.d_E > 0.)
584  deck.d_matData.d_K =
585  deck.d_matData.toK(deck.d_matData.d_E, deck.d_matData.d_nu);
586 
587  if (deck.d_matData.d_K > 0. && deck.d_matData.d_E < 0.)
588  deck.d_matData.d_E =
589  deck.d_matData.toE(deck.d_matData.d_K, deck.d_matData.d_nu);
590 
591  if (deck.d_matData.d_Gc > 0.)
592  deck.d_matData.d_KIc = deck.d_matData.toKIc(
593  deck.d_matData.d_Gc, deck.d_matData.d_nu, deck.d_matData.d_E);
594 
595  if (deck.d_matData.d_KIc > 0. && deck.d_matData.d_Gc < 0.)
596  deck.d_matData.d_Gc = deck.d_matData.toGc(
597  deck.d_matData.d_KIc, deck.d_matData.d_nu, deck.d_matData.d_E);
598 
599  // compute lame parameter
600  deck.d_matData.d_lambda =
601  deck.d_matData.toLambdaE(deck.d_matData.d_E, deck.d_matData.d_nu);
602  deck.d_matData.d_G =
603  deck.d_matData.toGE(deck.d_matData.d_E, deck.d_matData.d_nu);
604  deck.d_matData.d_mu = deck.d_matData.d_G;
605 
606  // get moment of influence function
607  double M = getMoment(dim);
608 
609  // compute peridynamic parameters
610  if (dim == 2) {
611  d_C = M_PI * deck.d_matData.d_Gc / (4. * M);
612  d_beta = 4. * deck.d_matData.d_lambda / (d_C * M);
613  } else if (dim == 3) {
614  d_C = 2. * deck.d_matData.d_Gc / (3. * M);
615  d_beta = 5. * deck.d_matData.d_lambda / (d_C * M);
616  }
617 
618  d_rbar = std::sqrt(0.5 / d_beta);
619  };
