30 struct BoxPartition {
31 
32  util::Point d_xc; // centroid of box
33  std::pair<util::Point, util::Point> d_box; // two corner points
34  double d_r; // radius of circle inscribing box
35  std::vector<size_t> d_nodes; // id of nodes in this box
36 
37  BoxPartition() : d_xc(util::Point()),
38  d_box({util::Point(), util::Point()}),
39  d_r(0.) {}
40 
41  bool isNear(const BoxPartition &box, const double &tol) const {
42  auto dx = d_xc - box.d_xc;
43  return dx.length() < d_r + box.d_r + tol;
44  }
45 
46  bool isNear(const util::Point &x, const double &tol) const {
47 
48  if (util::isLess(x.d_x, d_box.first.d_x - tol) or
49  util::isLess(x.d_y, d_box.first.d_y - tol) or
50  util::isLess(x.d_z, d_box.first.d_z - tol) or
51  util::isGreater(x.d_x, d_box.second.d_x + tol) or
52  util::isGreater(x.d_y, d_box.second.d_y + tol) or
53  util::isGreater(x.d_z, d_box.second.d_z + tol))
54  return false;
55 
56  return true;
57  }
58 
59  void addNode(const size_t &i) {
60  for (const auto &j : d_nodes)
61  if (j == i)
62  return;
63 
64  d_nodes.push_back(i);
65  }
66 };
