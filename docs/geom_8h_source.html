355 template <class T>
356 inline T l2Dist(const std::vector<T> &x1, const std::vector<T> &x2) {
357 
358  if (x1.size() != x2.size())
359  throw std::invalid_argument("size of two vectors not matching");
360 
361  T sum = 0.;
362  for (size_t i=0; i<x1.size(); i++)
363  sum += (x1[i] - x2[i]) * (x1[i] - x2[i]);
364 
365  return std::sqrt(sum);
366 }
