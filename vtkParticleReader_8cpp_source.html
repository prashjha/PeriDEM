44void rw::reader::VtkParticleReader::readNodes(
45 model::ModelData *model) {
46
47 if (model->d_x.size() == 0)
48 return;
49
50 //
51 // For nodes, we read following data
52 // 1. nodes current configuration
53 // 2. nodes displacement (if not then compute from current and reference
54 // config)
55 // 3. nodes velocity (if not then produce error)
56 //
57 d_grid_p = d_reader_p->GetOutput();
58 vtkIdType num_nodes = d_grid_p->GetNumberOfPoints();
59
60 if (num_nodes != model->d_x.size()) {
61 std::cerr << "Error: Number of points data = " << num_nodes <<
62 " in file does not match the number of points = "
63 << model->d_x.size() << ". Aborting reading.\n";
64 exit(1);
65 }
66
67 // read current configuration
68 for (size_t i = 0; i < num_nodes; i++) {
69 vtkIdType id = i;
70
71 double x[3];
72 d_grid_p->GetPoint(id, x);
73
74 util::Point i_node = util::Point(x[0], x[1], x[2]);
75 model->setX(i, i_node);
76 }
77
78 // read point field data
79 vtkPointData *p_field = d_grid_p->GetPointData();
80
81 std::vector<std::string> tags = {"Displacement", "Velocity"};
82 for (auto tag: tags) {
83
84 if (p_field->HasArray(tag.c_str())) {
85 vtkDataArray *array = p_field->GetArray(tag.c_str());
86
87 auto data_a = vtkSmartPointer<vtkDoubleArray>::New();
88 data_a->SetNumberOfComponents(3);
89 data_a->Allocate(3, 1); // allocate memory
90
91 for (size_t i = 0; i < array->GetNumberOfTuples(); i++) {
92 array->GetTuples(i, i, data_a);
93 auto xi = util::Point(data_a->GetValue(0), data_a->GetValue(1),
94 data_a->GetValue(2));
95 if (tag == "Displacement")
96 model->setU(i, xi);
97 else if (tag == "Velocity")
98 model->setV(i, xi);
99 }
100 } else {
101
102 if (tag == "Velocity") {
103 std::cerr << "Error: Restart file does not have <Velocity> data. "
104 "Aborting reading.\n";
105 exit(1);
106 } else if (tag == "Displacement") {
107 // compute from current and reference configuration
108 for (size_t i=0; i<model->d_x.size(); i++)
109 model->setU(i, model->getX(i) - model->getXRef(i));
110 }
111 }
112 }
113}
