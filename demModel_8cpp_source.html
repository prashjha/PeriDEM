289 void model::DEMModel::integrate() {
290 
291  // perform output at the beginning
292  if (d_n == 0 && d_outputDeck_p->d_performOut) {
293  log(fmt::format("DEMModel: Output step = {}, time = {:.6f} \n", d_n, d_time),
294  2);
295  output();
296  }
297 
298  // apply initial condition
299  if (d_n == 0)
300  applyInitialCondition();
301 
302  // apply loading
303  computeExternalDisplacementBC();
304  computeForces();
305 
306  for (size_t i = d_n; i < d_modelDeck_p->d_Nt; i++) {
307 
308  log(fmt::format("DEMModel: Time step: {}, time: {:8.6f}, steps completed = {}%\n",
309  i,
310  d_time,
311  float(i) * 100. / d_modelDeck_p->d_Nt),
312  2, d_n % d_infoN == 0, 3);
313 
314  clock_begin = steady_clock::now();
315  log("Integrating\n", false, 0, 3);
316  integrateStep();
317  double integrate_time =
318  util::methods::timeDiff(clock_begin, steady_clock::now());
319  integrate_compute_time += integrate_time;
320 
321  log(fmt::format(" Integration time (ms) = {}\n", integrate_time), 2, d_n % d_infoN == 0, 3);
322 
323  if (d_pDeck_p->d_testName == "two_particle") {
324 
325  // compute location of maximum shear stress and also compute
326  // penetration length
327  auto msg = ppTwoParticleTest();
328  log(msg, 2, d_n % d_infoN == 0, 3);
329  } else if (d_pDeck_p->d_testName == "compressive_test") {
330  auto msg = ppCompressiveTest();
331  log(msg, 2, d_n % d_infoN == 0, 3);
332  }
333 
334  // handle general output
335  if ((d_n % d_outputDeck_p->d_dtOut == 0) &&
336  (d_n >= d_outputDeck_p->d_dtOut) && d_outputDeck_p->d_performOut) {
337  clock_begin = steady_clock::now();
338  output();
339  }
340 
341  // check for stop
342  checkStop();
343 
344  } // loop over time steps
345 
346  log(fmt::format(
347  "DEMModel: Total compute time information (s) \n"
348  " {:22s} = {:8.2f} \n"
349  " {:22s} = {:8.2f} \n"
350  " {:22s} = {:8.2f} \n"
351  " {:22s} = {:8.2f} \n"
352  " {:22s} = {:8.2f} \n",
353  "Time integration", integrate_compute_time * 1.e-6,
354  "Peridynamics force", pd_compute_time * 1.e-6,
355  "Contact force", contact_compute_time * 1.e-6,
356  "Search tree update", tree_compute_time * 1.e-6,
357  "External force", extf_compute_time * 1.e-6)
358  );
359 }

368 void model::DEMModel::integrateCD() {
369 
370  // update velocity and displacement
371  d_currentDt = d_modelDeck_p->d_dt;
372  const auto dim = d_modelDeck_p->d_dim;
373 
374  tf::Executor executor(util::parallel::getNThreads());
375  tf::Taskflow taskflow;
376 
377  // update current position, displacement, and velocity of nodes
378  taskflow.for_each_index(
379  (std::size_t) 0, d_fPdCompNodes.size(), (std::size_t) 1,
380  [this, dim](std::size_t II) {
381  auto i = this->d_fPdCompNodes[II];
382 
383  const auto rho = this->getDensity(i);
384  const auto &fix = this->d_fix[i];
385 
386  for (int dof = 0; dof < dim; dof++) {
387  if (util::methods::isFree(fix, dof)) {
388  this->d_v[i][dof] += (this->d_currentDt / rho) * this->d_f[i][dof];
389  this->d_u[i][dof] += this->d_currentDt * this->d_v[i][dof];
390  this->d_x[i][dof] += this->d_currentDt * this->d_v[i][dof];
391  }
392  }
393 
394  this->d_vMag[i] = this->d_v[i].length();
395  } // loop over nodes
396  ); // for_each
397 
398  executor.run(taskflow).get();
399 
400  // advance time
401  d_n++;
402  d_time += d_currentDt;
403 
404  // update displacement bc
405  computeExternalDisplacementBC();
406 
407  // compute force
408  computeForces();
409 }

589

591  const bool is_state = d_particlesListTypeAll[0]->getMaterial()->isStateActive();
592 
593  // compute state-based helper quantities

595 
596  tf::Executor executor(util::parallel::getNThreads());
597  tf::Taskflow taskflow;
598 
599  taskflow.for_each_index(
600  (std::size_t) 0, d_fPdCompNodes.size(), (std::size_t) 1, [this](std::size_t II) {
601  auto i = this->d_fPdCompNodes[II];
602 
603  const auto rho = this->getDensity(i);
604  const auto &fix = this->d_fix[i];
605  const auto &ptId = this->getPtId(i);
606  auto &pi = this->getParticleFromAllList(ptId);
607 
608  if (pi->d_material_p->isStateActive()) {
609 
610  const double horizon = pi->getHorizon();
611  const double mesh_size = pi->getMeshSize();
612  const auto &xi = this->d_xRef[i];
613  const auto &ui = this->d_u[i];

640  this->d_fracture_p->setBondState(i, k, fs);
