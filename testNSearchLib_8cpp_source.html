39void stats(const std::vector<double> &x, double &mean, double &std) {
40 double mu = 0.;
41 for (auto &y : x)
42 mu += y;
43 mu = mu / x.size();
44
45 double s = 0.;
46 for (auto &y : x)
47 s += (y - mu) * (y - mu);
48 s = s / x.size();
49
50 std = std::sqrt(s);
51 mean = mu;
52}

54void lattice(double L, size_t Nx, size_t Ny, size_t Nz, double dL, int seed,
55 std::vector<util::Point> &x) {
56
57 RandGenerator gen(util::get_rd_gen(seed));
58 UniformDistribution dist(-dL, dL);
59
60 x.resize(Nx * Ny * Nz);
61
62 size_t count = 0;
63 for (size_t i = 0; i < Nx; i++)
64 for (size_t j = 0; j < Ny; j++)
65 for (size_t k = 0; k < Nz; k++) {
66 auto y = util::Point();
67 y.d_x = i * L + dist(gen);
68 y.d_y = j * L + dist(gen);
69 y.d_z = k * L + dist(gen);
70
71 x[count] = y;
72 count++;
73 }
74}

76template <class NSearch>
77double neighSearchTree(const std::vector<util::Point> &x,
78 const std::unique_ptr<NSearch> &nsearch, const double &r,
79 std::vector<std::vector<size_t>> &neigh,
80 std::vector<std::vector<float>> &neigh_sq_dist) {
81 neigh.resize(x.size());
82 neigh_sq_dist.resize(x.size());
83 auto t1 = steady_clock::now();
84
85 tf::Executor executor(util::parallel::getNThreads());
86 tf::Taskflow taskflow;
87
88 taskflow.for_each_index((std::size_t) 0, x.size(), (std::size_t) 1, [&x, &neigh, &neigh_sq_dist, &nsearch, r](std::size_t i) {
89 std::vector<int> neighs;
90 std::vector<float> sqr_dist;
91
92 if (nsearch->radiusSearch(x[i], r, neighs, sqr_dist) >
93 0) {
94 for (std::size_t j = 0; j < neighs.size(); ++j)
95 if (size_t(neighs[j]) != i) {
96 neigh[i].push_back(size_t(neighs[j]));
97 neigh_sq_dist[i].push_back(sqr_dist[j]);
98 }
99 }
100 }
101 ); // for_each
102
103 executor.run(taskflow).get();
104
105 auto t2 = steady_clock::now();
106 return util::methods::timeDiff(t1, t2, "microseconds");
107}

142double neighSearchBrute(const std::vector<util::Point> &x, const double &r,
143 std::vector<std::vector<size_t>> &neigh,
144 std::vector<std::vector<float>> &neigh_sq_dist) {
145 neigh.resize(x.size());
146 neigh_sq_dist.resize(x.size());
147
148 auto t1 = steady_clock::now();
149
150 tf::Executor executor(util::parallel::getNThreads());
151 tf::Taskflow taskflow;
152
153 taskflow.for_each_index((std::size_t) 0, x.size(), (std::size_t) 1, [&x, &neigh, &neigh_sq_dist, r](std::size_t i) {
154 auto searchPoint = x[i];
155
156 for (size_t j = 0; j < x.size(); j++) {
157 auto dx = searchPoint - x[j];
158 auto l = dx.length();
159 if (util::isLess(l, r) and j != i) {
160 neigh[i].push_back(j);
161 neigh_sq_dist[i].push_back(l);
162 }
163 }
164 }
165 ); // for_each
166
167 executor.run(taskflow).get();
168
169 auto t2 = steady_clock::now();
170 return util::methods::timeDiff(t1, t2, "microseconds");
171}

173std::string compare_results(const std::vector<std::vector<size_t>> &neigh1,
174 const std::vector<std::vector<size_t>> &neigh2,
175 std::vector<std::string> tags,
176 int check_nodes_num = -1,
177 bool only_err_count = false) {
178 size_t error_size = 0;
179 size_t error_nodes = 0;
180 size_t error_neighs = 0;
181 std::ostringstream composs;
182 for (size_t i = 0; i < neigh1.size(); i++) {
183
184 if (check_nodes_num > 0 and i > check_nodes_num)
185 continue;
186
187 size_t err_neighs = 0;
188 auto &n1 = neigh1[i];
189 auto &n2 = neigh2[i];
190
191 bool header_done = false;
192 if (n1.size() != n2.size()) {
193 composs << " Node = " << i << " \n";
194 composs << fmt::format(" size ({}) {} != {} ({}) not matching\n",
195 tags[0], n1.size(), n2.size(), tags[1]);
196 header_done = true;
197 error_size++;
198 }
199
200 for (auto j : n2) {
201 if (!isInList(&n1, j)) {
202 if (!header_done)
203 composs << " Node = " << i << " \n";
204
205 composs << fmt::format(" neigh = {} in {} search not found in {} "
206 "search neighs list\n",
207 j, tags[1], tags[0]);
208 err_neighs += 1;
209 }
210 }
211
212 if (err_neighs > 0)
213 error_neighs += err_neighs;
214 }
215
216 if (only_err_count)
217 return fmt::format(" error_size = {}, error_neighs = {}\n", error_size,
218 error_neighs);
219 else
220 return fmt::format(" error_size = {}, error_neighs = {}\n", error_size,
221 error_neighs) +
222 composs.str();
223}

226std::string test::testNanoflann(size_t N, double L, double dL, int seed) {
227
228 // create 3D lattice and perturb each lattice point
229 size_t Nx, Ny, Nz;
230 Nx = Ny = Nz = N;
231 size_t N_tot = Nx * Ny * Nz;
232
233 std::vector<util::Point> x(N_tot, util::Point());
234 lattice(L, Nx, Ny, Nz, dL, seed, x);
235 std::cout << "Total points = " << x.size() << "\n";
236
237 std::vector<std::vector<size_t>> neigh_nflann(N_tot, std::vector<size_t>());
238 std::vector<std::vector<float>> neigh_nflann_sq_dist(N_tot,
239 std::vector<float>());
240
241 std::vector<std::vector<size_t>> neigh_brute(N_tot, std::vector<size_t>());
242 std::vector<std::vector<float>> neigh_brute_sq_dist(N_tot,
243 std::vector<float>());
244
245 std::vector<std::vector<size_t>> neigh_pcl(N_tot, std::vector<size_t>());
246 std::vector<std::vector<float>> neigh_pcl_sq_dist(N_tot,
247 std::vector<float>());
248
249 //
250 // brute-force search
251 //
252 double search_r = 1.5 * L;
253 auto brute_force_search_time =
254 neighSearchBrute(x, search_r, neigh_brute, neigh_brute_sq_dist);
255
256 //
257 // nanoflann tree search
258 //
259 std::unique_ptr<nsearch::NFlannSearchKd> nflann_nsearch = std::make_unique<nsearch::NFlannSearchKd>(x, 0);
260 auto nflann_tree_set_time = nflann_nsearch->updatePointCloud(x, true);
261 nflann_tree_set_time += nflann_nsearch->setInputCloud();
262 auto nflann_tree_search_time =
263 neighSearchTreeSizet(x, nflann_nsearch, search_r, neigh_nflann, neigh_nflann_sq_dist);
264
265 //
266 // Compare three search lists
267 //
268 auto nflann_brute_compare = compare_results(
269 neigh_nflann, neigh_brute, {"nflann_tree", "brute_force"}, -1, true);
270
271 std::ostringstream msg;
272 msg << fmt::format(" Setup times (microseconds): \n"
273 " nflann_tree_set_time = {}\n",
274 nflann_tree_set_time);
275
276 msg << fmt::format(" Search times (microseconds): \n"
277 " brute_force_search_time = {}\n"
278 " nflann_tree_search_time = {}\n",
279 brute_force_search_time,
280 nflann_tree_search_time);
281
282 msg << fmt::format(" Comparison results: \n"
283 " nflann_brute_compare: \n{}\n",
284 nflann_brute_compare);
285
286 return msg.str();
287}
