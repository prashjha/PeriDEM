233 const util::Point &searchPoint, const double &search_r,
234 std::vector<int> &neighs,
235 std::vector<float> &sqr_dist) override {
236
237 // ugly but quick fix
238 // first, get results using int and float and then convert
239 std::vector<size_t> neighs_temp;
240 std::vector<double> sqr_dist_temp;
241 auto N =
242 this->radiusSearch(searchPoint, search_r, neighs_temp, sqr_dist_temp);
243
244 if (N > 0) {
245 neighs.resize(N);
246 sqr_dist.resize(N);
247 for (size_t i=0; i<N; i++) {
248 neighs.push_back(int(neighs_temp[i]));
249 sqr_dist.push_back(float(sqr_dist_temp[i]));
250 }
251 }
252
253 return N;
254 };
