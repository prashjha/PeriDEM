100  d_geom_p(geom),
101  d_tform(transform),
102  d_mesh_p(mesh),
103  d_pRadius(geom->boundingRadius()) {
104 
105  if (d_type == "particle")
106  d_typeIndex = 0;
107  else if (d_type == "wall")
108  d_typeIndex = 1;
109 
110  d_computeForce = !d_allDofsConstrained;
111 
112  // check
113  if (d_type == "particle" and d_isWall) {
114  std::cerr << "Error: Can not have d_type = 'particle' and d_isWall = true.\n";
115  exit(EXIT_FAILURE);
116  }
117  if (d_type == "wall" and !d_isWall) {
118  std::cerr << "Error: Can not have d_type = 'wall' and d_isWall = false.\n";
119  exit(EXIT_FAILURE);
120  }
121 
122  if (populate_data) {
123 
124  d_globStart = d_modelData_p->d_x.size();
125  d_globEnd = d_modelData_p->d_x.size() + d_rp_p->getNumNodes();
126  for (size_t i = 0; i < d_rp_p->getNumNodes(); i++) {
127 
128  d_modelData_p->d_xRef.push_back(d_tform.apply(d_rp_p->getNode(i)));
129  d_modelData_p->d_x.push_back(d_tform.apply(d_rp_p->getNode(i)));
130  d_modelData_p->d_u.push_back(util::Point());
131  d_modelData_p->d_v.push_back(util::Point());
132  d_modelData_p->d_f.push_back(util::Point());
133  d_modelData_p->d_vol.push_back(
134  d_rp_p->getNodalVolume(i) *
135  std::pow(d_tform.d_scale, d_rp_p->getDimension()));
136  d_modelData_p->d_fix.push_back(uint8_t(0));
137  d_modelData_p->d_forceFixity.push_back(uint8_t(0));
138  d_modelData_p->d_thetaX.push_back(0.);
139  d_modelData_p->d_mX.push_back(0.);
140  d_modelData_p->d_ptId.push_back(id); // id of this particle
141  }
142  }
143 
144  d_h = util::computeMeshSize(d_modelData_p->d_x, d_globStart,
145  d_globEnd);
146 
147  // initialize material class
148  inp::MaterialDeck p_material_deck = material_deck;
149  // double horizon = d_geom_p->inscribedRadius();
150  double horizon = p_material_deck.d_horizon;
151  if (p_material_deck.d_horizonMeshRatio > 0.)
152  horizon = p_material_deck.d_horizonMeshRatio * d_h;
153 
154  if (p_material_deck.d_materialType == "RNPBond")
155  d_material_p = std::make_unique<material::RnpMaterial>(
156  p_material_deck, d_rp_p->getDimension(), horizon);
157  else if (p_material_deck.d_materialType == "PMBBond")
158  d_material_p = std::make_unique<material::PmbMaterial>(
159  p_material_deck, d_rp_p->getDimension(), horizon);
160  else if (p_material_deck.d_materialType == "PDElasticBond")
161  d_material_p = std::make_unique<material::PdElastic>(
162  p_material_deck, d_rp_p->getDimension(), horizon);
163  else if (p_material_deck.d_materialType == "PDState")
164  d_material_p = std::make_unique<material::PdState>(
165  p_material_deck, d_rp_p->getDimension(), horizon);
166 
167  d_horizon = horizon;
168  d_density = d_material_p->getDensity();
169 
170  // d_material_p->print();
171 
172  // set contact radius for internal contact
173  d_Rc = 0.95 * d_h;
174 
175  // set contact coefficient for internal contact
176  d_Kn = (18. / (M_PI * std::pow(horizon, 5))) *
177  d_material_p->computeMaterialProperties(getDimension()).d_K;
178 
179  if (!d_computeForce) {
180  std::cout << "Warning: Compute force is OFF in particle with id = "
181  << d_id << "\n";
182  }
183  if (d_allDofsConstrained) {
184  std::cout << "Warning: All DoFs are OFF in particle with id = "
185  << d_id << "\n";
186  }
187 }
