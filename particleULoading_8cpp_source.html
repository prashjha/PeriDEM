41 bool loading::ParticleULoading::needToProcessParticle(size_t id, const inp::PBCData &bc) {
42  // if there is a list, and if particle is not in the list, skip
43  bool skip_condition1 = (bc.d_selectionType == "particle"
44  || bc.d_selectionType == "region_with_include_list")
45  && !isInList(id, bc.d_pList);
46  // if there is an exclusion list, and if particle is in the list, skip
47  bool skip_condition2 = (bc.d_selectionType == "region_with_exclude_list")
48  && isInList(id, bc.d_pNotList);
49  // if there is a inclusion and an exclusion list,
50  // and if particle is either in the exclusion list or not in the inclusion list, skip
51  bool skip_condition3 = (bc.d_selectionType == "region_with_include_list_with_exclude_list")
52  && (isInList(id, bc.d_pNotList) ||
53  !isInList(id, bc.d_pList));
54 
55  bool skip = skip_condition1 or skip_condition2 or skip_condition3;
56  return !skip;
57 }

59 bool loading::ParticleULoading::needToComputeDof(const util::Point &x,
60  size_t id,
61  const inp::PBCData &bc) {
62  if (bc.d_selectionType == "region" && bc.d_regionGeom_p->isInside(x))
63  return true;
64  else if (bc.d_selectionType == "region_with_include_list" &&
65  bc.d_regionGeom_p->isInside(x) &&
66  isInList(id, bc.d_pList))
67  return true;
68  else if (bc.d_selectionType == "region_with_exclude_list" &&
69  bc.d_regionGeom_p->isInside(x) &&
70  !isInList(id, bc.d_pNotList))
71  return true;
72  else if (bc.d_selectionType == "region_with_include_list_with_exclude_list" &&
73  bc.d_regionGeom_p->isInside(x) &&
74  isInList(id, bc.d_pList) &&
75  !isInList(id, bc.d_pNotList))
76  return true;
77  else if (bc.d_selectionType == "particle" &&
78  isInList(id, bc.d_pList))
79  return true;
80 
81  return false;
82 }

112 void loading::ParticleULoading::apply(const double &time,
113  particle::BaseParticle *particle) {
114 
115  for (size_t s = 0; s < d_bcData.size(); s++) {
116 
117  // get alias for bc data
118  const auto &bc = d_bcData[s];
119 
120  // if this is zero displacement condition, and if we have already applied
121  // displacement then do not need to reapply this
122  if (bc.d_isDisplacementZero && d_pZeroDisplacementApplied[s])
123  continue;
124 
125  // set to true so that next time this is not called
126  if (bc.d_isDisplacementZero)
127  d_pZeroDisplacementApplied[s] = true;
128 
129  // check if we need to process this particle
130  if (!needToProcessParticle(particle->getId(), bc))
131  continue;
132 
133  // get bounding box
134  auto box = bc.d_regionGeom_p->box();
135 
136  // for (size_t i = 0; i < particle->getNumNodes(); i++) {
137  tf::Executor executor(util::parallel::getNThreads());
138  tf::Taskflow taskflow;
139 
140  taskflow.for_each_index(
141  (std::size_t) 0, particle->getNumNodes(), (std::size_t) 1,
142  [time, &particle, bc, box, this] (std::size_t i) {
143 
144  const auto x = particle->getXRefLocal(i);
145 
146  double umax = bc.d_timeFnParams[0];
147  double du = 0.;
148  double dv = 0.;
149 
150  if (needToComputeDof(x, particle->getId(), bc)) {
151 
152  // apply spatial function
153  if (bc.d_spatialFnType == "sin_x") {
154  double a = M_PI * bc.d_spatialFnParams[0];
155  umax = umax * std::sin(a * x.d_x);
156  } else if (bc.d_spatialFnType == "sin_y") {
157  double a = M_PI * bc.d_spatialFnParams[0];
158  umax = umax * std::sin(a * x.d_y);
159  } else if (bc.d_spatialFnType == "linear_x") {
160  double a = bc.d_spatialFnParams[0];
161  umax = umax * a * x.d_x;
162  } else if (bc.d_spatialFnType == "linear_y") {
163  double a = bc.d_spatialFnParams[0];
164  umax = umax * a * x.d_y;
165  }
166 
167  // apply time function
168  if (bc.d_timeFnType == "constant")
169  du = umax;
170  else if (bc.d_timeFnType == "linear") {
171  du = umax * time;
172  dv = umax;
173  } else if (bc.d_timeFnType == "quadratic") {
174  du = umax * time + bc.d_timeFnParams[1] * time * time;
175  dv = umax + bc.d_timeFnParams[1] * time;
176  } else if (bc.d_timeFnType == "sin") {
177  double a = M_PI * bc.d_timeFnParams[1];
178  du = umax * std::sin(a * time);
179  dv = umax * a * std::cos(a * time);
180  }
181 
182  auto u_i = util::Point();
183  auto v_i = util::Point();
184  for (auto d : bc.d_direction) {
185  u_i[d-1] = du;
186  v_i[d-1] = dv;
187  }
188 
189  if (bc.d_timeFnType == "rotation") {
190  auto x0 = util::Point(bc.d_timeFnParams[1], bc.d_timeFnParams[2],
191  bc.d_timeFnParams[3]);
192  auto dx = x - x0;
193  auto r_x = util::rotate2D(
194  dx, bc.d_timeFnParams[0] * time);
195  auto dr_x = util::derRotate2D(
196  dx, bc.d_timeFnParams[0] * time);
197 
198  u_i += r_x - dx;
199  v_i += bc.d_timeFnParams[0] * dr_x;
200  }
201 
202  for (auto d : bc.d_direction) {
203  particle->setULocal(i, d-1, u_i[d-1]);
204  particle->setVLocal(i, d-1, v_i[d-1]);
205  auto xref = particle->getXRefLocal(i)[d-1];
206  particle->setXLocal(i, d-1, u_i[d-1] + xref);
207  }
208  } // if compute displacement
209  }
210  ); // for_each
211 
212  executor.run(taskflow).get();
213  } // loop over bc sets
214 }
