115 template <typename T>
116 inline size_t maxIndex(const std::vector<T> &data,
117  size_t data_start, size_t data_end) {
118 
119  if (data.size() == 0) {
120  std::cerr << "Error: maxIndex() is called with data of size " << data.size()
121  << ".\n";
122  exit(EXIT_FAILURE);
123  }
124 
125  if (data_end == 0 or data_end > data.size()) {
126  std::cerr << "Error: maxIndex() data_end = " << data_end
127  << " is not valid for the data of size " << data.size()
128  << ".\n";
129  exit(EXIT_FAILURE);
130  }
131 
132  if (data_start > data.size() - 1) {
133  std::cerr << "Error: maxIndex() data_start = " << data_start
134  << " is not valid for the data of size " << data.size()
135  << ".\n";
136  exit(EXIT_FAILURE);
137  }
138 
139  // initialize original index locations
140  std::vector<size_t> idx(data_end - data_start);
141  std::iota(idx.begin(), idx.end() , 0);
142 
143  std::stable_sort(idx.begin(), idx.end(),
144  [&data, &data_start](size_t i1, size_t i2)
145  {return data[i1 + data_start] > data[i2 + data_start];});
146 
147  return idx[0] + data_start;
148 };
