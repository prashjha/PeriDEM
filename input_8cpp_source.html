75  }

81  std::cerr << "Error: To define complex geometry, we require vector of "
82  "flags in yaml data block = "
83  << data_block_name << "\n";
84  exit(EXIT_FAILURE);
85  }
86  }
87 
88  params.clear();
89  if (data["Parameters"]) {
90  for (auto f : data["Parameters"])
91  params.push_back(f.as<double>());
92  } else {
93  std::cerr << "Error: Parameters defining geometry is not present in data block = "

165 void inp::Input::setModelDeck() {
166  d_modelDeck_p = std::make_shared<inp::ModelDeck>();
167 
168  if (d_inputFilename.empty() and d_createDefault)
169  return;
170 
171  YAML::Node config = YAML::LoadFile(d_inputFilename);
172 
173  // read dimension
174  if (config["Model"]["Dimension"])
175  d_modelDeck_p->d_dim = config["Model"]["Dimension"].as<size_t>();
176  else {
177  std::cerr << "Error: Please specify the dimension.\n";
178  exit(1);
179  }
180 
181  // read discretization info
182  if (config["Model"]["Discretization_Type"]["Time"])
183  d_modelDeck_p->d_timeDiscretization =
184  config["Model"]["Discretization_Type"]["Time"].as<std::string>();
185 
186  if (config["Model"]["Discretization_Type"]["Spatial"])
187  d_modelDeck_p->d_spatialDiscretization =
188  config["Model"]["Discretization_Type"]["Spatial"].as<std::string>();
189 
190 
191  if (d_modelDeck_p->d_timeDiscretization == "central_difference" or
192  d_modelDeck_p->d_timeDiscretization == "velocity_verlet")
193  d_modelDeck_p->d_simType = "explicit";
194 
195  if (config["Model"]["Populate_ElementNodeConnectivity"])
196  d_modelDeck_p->d_populateElementNodeConnectivity = true;
197 
198  if (config["Model"]["Quad_Approximation_Order"])
199  d_modelDeck_p->d_quadOrder = config["Model"]["Quad_Approximation_Order"].as<size_t>();
200 
201  // check if this is peridem simulation
202  if (config["Particle"])
203  d_modelDeck_p->d_isPeriDEM = true;
204 
205  // read horizon and horizon to mesh ratio (if available)
206  if (config["Model"]["Horizon"])
207  d_modelDeck_p->d_horizon = config["Model"]["Horizon"].as<double>();
208  if (config["Model"]["Horizon_h_Ratio"])
209  d_modelDeck_p->d_rh = config["Model"]["Horizon_h_Ratio"].as<int>();
210  if (config["Model"]["Mesh_Size"])
211  d_modelDeck_p->d_h = config["Model"]["Mesh_Size"].as<double>();
212 
213  if (!config["Model"]["Horizon"]) {
214 
215  // check and issue error if this is not peridem simulation
216  if (!d_modelDeck_p->d_isPeriDEM)
217  if (!config["Model"]["Horizon_h_Ratio"] or
218  !config["Model"]["Mesh_Size"]) {
219 
220  std::cerr << "Error: Horizon is not provided. In this case "
221  "Horizon_h_Ratio and Mesh_Size are necessary to compute "
222  "horizon.\n";
223  exit(1);
224  }
225 
226  d_modelDeck_p->d_horizon = d_modelDeck_p->d_h * double(d_modelDeck_p->d_rh);
227  }
228 
229  if (!config["Model"]["Mesh_Size"])
230  if (config["Model"]["Horizon_h_Ratio"])
231  d_modelDeck_p->d_h =
232  d_modelDeck_p->d_horizon / double(d_modelDeck_p->d_rh);
233 
234  // read final time and time step
235  if (config["Model"]["Final_Time"])
236  d_modelDeck_p->d_tFinal = config["Model"]["Final_Time"].as<double>();
237  if (config["Model"]["Time_Steps"])
238  d_modelDeck_p->d_Nt = config["Model"]["Time_Steps"].as<int>();
239 
240  if (std::abs(d_modelDeck_p->d_tFinal) < 1.0E-10 or d_modelDeck_p->d_Nt <= 0) {
241  std::cerr << "Error: Check Final_Time and Time_Steps data.\n";
242  exit(1);
243  }
244 
245  d_modelDeck_p->d_dt = d_modelDeck_p->d_tFinal / d_modelDeck_p->d_Nt;
246 
247  // check if this is restart problem
248  if (config["Restart"])
249  d_modelDeck_p->d_isRestartActive = true;
250 
251  // check if this is restart problem
252  if (config["Model"]["Seed"])
253  d_modelDeck_p->d_seed = config["Model"]["Seed"].as<int>();;
254 } // setModelDeck

256 void inp::Input::setParticleDeck() {
257  d_particleDeck_p = std::make_shared<inp::ParticleDeck>();
258 
259  if (d_inputFilename.empty() and d_createDefault)
260  return;
261 
262  YAML::Node config = YAML::LoadFile(d_inputFilename);
263 
264  // <<<< >>>>
265  // <<<< STEP 1 - Read container geometry details >>>>
266  // <<<< >>>>
267  // read particle container geometry and size
268  if (config["Container"]["Geometry"]) {
269  auto ce = config["Container"]["Geometry"];
270 
271  readGeometry(ce,
272  "Container->Geometry",
273  d_particleDeck_p->d_contGeomName,
274  d_particleDeck_p->d_contGeomParams,
275  d_particleDeck_p->d_contGeomComplexInfo);
276 
277  util::geometry::createGeomObject(
278  d_particleDeck_p->d_contGeomName,
279  d_particleDeck_p->d_contGeomParams,
280  d_particleDeck_p->d_contGeomComplexInfo.first,
281  d_particleDeck_p->d_contGeomComplexInfo.second,
282  d_particleDeck_p->d_contGeom_p,
283  d_modelDeck_p->d_dim);
284  //std::cout << d_particleDeck_p->d_contGeom_p->printStr(0, 0) << std::flush;
285  } else {
286  std::cerr << "Error: Need particle container information.\n";
287  exit(1);
288  }
289 
290  // <<<< >>>>
291  // <<<< STEP 2 - Read Zone data >>>>
292  // <<<< >>>>
293  // read input file
294  if (!config["Zone"]) {
295  std::cerr << "Error: Zone information is not provided.\n";
296  exit(1);
297  }
298  auto n = config["Zone"]["Zones"].as<size_t>();
299  if (n == 0) {
300  std::cerr << "Error: Require at least one zone to define particles.\n";
301  exit(1);
302  }
303 
304  d_particleDeck_p->d_zoneVec.resize(n);
305  d_particleDeck_p->d_particleZones.resize(n);
306  d_particleDeck_p->d_zoneToParticleORWallDeck.resize(n);
307 
308  // read zone and particle information
309  for (size_t z = 1; z <= n; z++) {
310 
311  std::string read_zone = "Zone_";
312  read_zone.append(std::to_string(z));
313 
314  auto e = config["Zone"][read_zone];
315 
316  // <<<< >>>>
317  // <<<< STEP 2.1 - Read this zone information >>>>
318  // <<<< >>>>
319  auto zone_data = inp::Zone();
320  zone_data.d_zoneId = z - 1;
321  setZoneData({"Zone", read_zone}, &zone_data);
322 
323  // store zone data
324  d_particleDeck_p->d_zoneVec[z-1] = zone_data;
325 
326  // <<<< >>>>
327  // <<<< STEP 2.2 - Read particle information in this zone >>>>
328  // <<<< >>>>
329  auto particle_data = inp::ParticleZone();
330  particle_data.d_zone = zone_data;
331  particle_data.d_isWall = false;
332  if (e["Is_Wall"])
333  particle_data.d_isWall = e["Is_Wall"].as<bool>();
334 
335  setParticleData(read_zone, &particle_data);
336 
337  // <<<< >>>>
338  // <<<< STEP 2.3 - Read mesh and reference particle information >>>>
339  // <<<< >>>>
340  setZoneMeshDeck({"Mesh", read_zone}, &(particle_data.d_meshDeck));
341 
342  // <<<< >>>>
343  // <<<< STEP 2.4 - Read material properties of this zone >>>>
344  // <<<< >>>>
345  setZoneMaterialDeck({"Material", read_zone}, &(particle_data.d_matDeck),
346  z);
347 
348  // add to the list
349  d_particleDeck_p->d_particleZones[z-1] = particle_data;
350 
351  // update zone to particle/wall deck map
352  d_particleDeck_p->d_zoneToParticleORWallDeck[z - 1] =
353  std::make_pair(particle_data.d_isWall ? "wall" : "particle", z - 1);
354  } // read zone and particle information
355 
356  // <<<< >>>>
357  // <<<< STEP 3 - Read neighbor search data >>>>
358  // <<<< >>>>
359  if (config["Neighbor"]) {
360  auto ne = config["Neighbor"];
361  if (ne["Update_Criteria"])
362  d_particleDeck_p->d_pNeighDeck.d_updateCriteria =
363  ne["Update_Criteria"].as<std::string>();
364 
365  if (ne["Search_Factor"])
366  d_particleDeck_p->d_pNeighDeck.d_sFactor =
367  ne["Search_Factor"].as<double>();
368 
369  for (auto p : d_particleDeck_p->d_particleZones) {
370 
371  // if particle is actually a wall then we do it differently
372  double tol_check = 0.;
373  if (p.d_isWall) {
374 
375  // check if wall has geometry object and it is not a null object
376  if (p.d_geom_p->d_name != "null" and !p.d_geom_p->d_name.empty()) {
377  if (p.d_geomParams.empty()) {
378  std::cerr
379  << "Error: Cannot set neighbor list data as the geometrical "
380  "parameter for the particle/wall is missing.\n";
381  exit(EXIT_FAILURE);
382  }
383  tol_check = p.d_geomParams[0];
384  } else {
385  // use container geometry or zone geometry to get the tolerance parameter
386  if (p.d_zone.d_geomParams.empty()) {
387  std::cerr
388  << "Error: Cannot set neighbor list data as the geometrical "
389  "parameter the zone this wall/particle belongs to is missing.\n";
390  exit(EXIT_FAILURE);
391  }
392  tol_check = p.d_zone.d_geomParams[0];
393  } // if else for wall geometry
394  } else {
395  if (p.d_geomParams.empty()) {
396  std::cerr
397  << "Error: Cannot set neighbor list data as the geometrical "
398  "parameter for the particle is missing.\n";
399  exit(EXIT_FAILURE);
400  }
401  tol_check = p.d_geomParams[0];
402  }
403 
404  if (util::isGreater(tol_check, d_particleDeck_p->d_pNeighDeck.d_sTol))
405  d_particleDeck_p->d_pNeighDeck.d_sTol = tol_check;
406  }
407 
408  // multiply by the search factor
409  d_particleDeck_p->d_pNeighDeck.d_sTol *=
410  d_particleDeck_p->d_pNeighDeck.d_sFactor;
411  } else {
412  std::cerr << "Error: Need neighbor list information.\n";
413  exit(1);
414  }
415 
416  // <<<< >>>>
417  // <<<< STEP 4 - Read gravity force information if any >>>>
418  // <<<< >>>>
419  if (config["Force_BC"]) {
420  if (config["Force_BC"]["Gravity"]) {
421 
422  d_particleDeck_p->d_gravityActive = true;
423 
424  auto gf = std::vector<double>(3, 0.);
425  size_t gf_c = 0;
426  for (auto e : config["Force_BC"]["Gravity"])
427  gf[gf_c++] = e.as<double>();
428 
429  d_particleDeck_p->d_gravity = util::Point(gf[0], gf[1], gf[2]);
430  }
431  }
432 
433  // <<<< >>>>
434  // <<<< STEP 5 - Read force bc and displacement bc >>>>
435  // <<<< >>>>
436  std::vector<std::string> vtags = {"Displacement_BC", "Force_BC"};
437 
438  for (const auto &tag : vtags) {
439 
440  // get number of force bc sets
441  int nsets = 0;
442  if (config[tag]["Sets"])
443  nsets = config[tag]["Sets"].as<int>();
444 
445  for (size_t s = 1; s <= nsets; s++) {
446 
447  // prepare string Set_s to read file
448  std::string read_set = "Set_";
449  read_set.append(std::to_string(s));

456  bc_deck.d_selectionType = e["Selection_Type"].as<std::string>();

460 
461  bc_deck.d_selectionType = "region";
462

464  bc_deck.d_selectionType += "_with_include_list";

472  bc_deck.d_selectionType += "_with_exclude_list";

562
